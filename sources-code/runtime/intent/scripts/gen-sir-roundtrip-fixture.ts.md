#!/usr/bin/env bun
/**
 * D-O5m.followup-3 Phase 2 — generate the cross-language SIR
 * roundtrip fixture.
 *
 * Reference: runtime/intent/scripts/gen-voice-fixture.ts
 *            (the Phase 1 sibling -- same fixture-generator pattern,
 *            same write-to-mobile-test-fixtures destination);
 *            apps/oddjobz-mobile/lib/src/voice/sir_extractor.dart
 *            (the Dart-side consumer that asserts byte-for-byte
 *            parity against this fixture);
 *            runtime/intent/src/__tests__/sir-roundtrip-fixture.test.ts
 *            (the TS-side parity test).
 *
 * The fixture is a {transcript, expected_intent, canonical_intent_
 * json} triple.  The expected_intent is what the on-device extractor
 * SHOULD produce given the transcript; canonical_intent_json is the
 * byte-exact JSON encoding the wire layer ships.  Phase 2's
 * load-bearing claim:
 *
 *     given the same transcript + extension grammar, both the TS
 *     reference encoder and the Dart on-device extractor produce
 *     the same canonical JSON bytes
 *
 * That parity is what makes the brain's L1 validation work the same
 * whether the Intent came from the phone or the brain itself.
 */

import { writeFileSync } from 'node:fs';
import { dirname, resolve } from 'node:path';
import { fileURLToPath } from 'node:url';

// The canonical ordering — matches sir_extractor.dart::canonicalise
// Intent and the TS Intent declaration order in
// runtime/intent/src/types.ts.
const ORDER: ReadonlyArray<string> = [
  'id',
  'correlationId',
  'companionOf',
  'summary',
  'category',
  'taxonomy',
  'action',
  'constraints',
  'target',
  'transferTo',
  'fulfillment',
  'confidence',
  'source',
  'producerMeta',
];

function canonicaliseIntent(input: Record<string, unknown>): Record<string, unknown> {
  const out: Record<string, unknown> = {};
  for (const k of ORDER) {
    if (Object.prototype.hasOwnProperty.call(input, k) && input[k] != null) {
      out[k] = input[k];
    }
  }
  for (const [k, v] of Object.entries(input)) {
    if (!Object.prototype.hasOwnProperty.call(out, k) && v != null) {
      out[k] = v;
    }
  }
  return out;
}

function encodeCanonicalIntent(input: Record<string, unknown>): string {
  return JSON.stringify(canonicaliseIntent(input));
}

// Phase 2 fixture — a tradie says "job 12345 done invoiced", the
// on-device extractor produces the trades-lexicon Intent below.
const transcript = 'job 12345 done invoiced';

// The Intent the extractor produces.  Keys are deliberately
// shuffled so the canonicaliser has work to do; the canonical JSON
// has them in declaration order.
const intentFromExtractor: Record<string, unknown> = {
  source: 'voice',
  confidence: 1.0,
  constraints: [
    { kind: 'state', requiredPhase: 'in_progress' },
  ],
  taxonomy: { what: 'jobs', how: 'transition', why: 'close-out' },
  category: { lexicon: 'trades', category: 'invoice' },
  action: 'invoice',
  summary: 'job 12345 done invoiced',
  id: 'sir-roundtrip-001',
  target: { objectId: '12345', typePath: 'oddjobz.job' },
};

const canonical = canonicaliseIntent(intentFromExtractor);
const canonicalJson = encodeCanonicalIntent(intentFromExtractor);

const fixture = {
  _comment:
    'Generated by runtime/intent/scripts/gen-sir-roundtrip-fixture.ts -- DO NOT EDIT BY HAND.',
  transcript,
  expectedIntent: canonical,
  canonicalIntentJson: canonicalJson,
  // Spell out the field-order claim explicitly so a future
  // regression has a one-grep failure mode.
  canonicalKeyOrder: Object.keys(canonical),
};

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

writeFileSync(
  resolve(__dirname, '../fixtures/sir-roundtrip-fixture.json'),
  JSON.stringify(fixture, null, 2) + '\n',
);
console.log(JSON.stringify(fixture, null, 2));
