// BRC-100 signed-request envelope parser/builder for the wallet bundle.
//
// Per WALLET-TIER-CUSTODY.md §8.2, BRC-100 wraps a request body with four
// wire-protocol headers:
//   x-brc100-identitykey  — 33-byte compressed secp256k1 pubkey, hex
//   x-brc100-nonce        — 32-byte random, hex
//   x-brc100-timestamp    — Unix seconds (string or number)
//   x-brc100-signature    — DER-encoded ECDSA signature over the canonical
//                            digest, hex (no sighash byte — pure DER)
//
// The dApp ↔ iframe channel uses the same envelope shape, just delivered as
// the body of a postMessage instead of HTTP headers. v0.1 verifies the
// signature and replay-window timestamp; the actual request/response shape
// is defined per-method (W9 will spell out the BRC-100 RPC surface).
//
// Canonical digest = SHA256( identityKey || nonce || timestamp_le8 || body ).
// `body` is whatever the dApp wants signed (a UTF-8 JSON string in v0.1).

import * as secp from '@noble/secp256k1';
import { sha256 as nobleSha256 } from '@noble/hashes/sha2';
import { decodeDer, encodeDer } from './der';

export interface Brc100Envelope {
  identityKey: Uint8Array; // 33 bytes
  nonce: Uint8Array; // 32 bytes
  timestamp: number; // Unix seconds
  signature: Uint8Array; // DER bytes, no sighash
  body: Uint8Array; // signed payload
}

export type Brc100ParseResult =
  | { ok: true; envelope: Brc100Envelope }
  | { ok: false; reason: string };

/** Default: reject envelopes with a timestamp older than 5 minutes (matches §8.3 OTP rate-limiting). */
const DEFAULT_REPLAY_WINDOW_SECS = 300;

function hexToBytes(hex: string): Uint8Array {
  if (hex.length % 2 !== 0) throw new Error('hex: odd length');
  const out = new Uint8Array(hex.length / 2);
  for (let i = 0; i < out.length; i++) {
    out[i] = parseInt(hex.slice(i * 2, i * 2 + 2), 16);
  }
  return out;
}

function bytesToHex(bytes: Uint8Array): string {
  let s = '';
  for (const b of bytes) s += b.toString(16).padStart(2, '0');
  return s;
}

/**
 * Compute the canonical digest the signature commits to.
 * = SHA256( identityKey(33) || nonce(32) || timestamp_le8 || body )
 */
export function envelopeDigest(env: Pick<Brc100Envelope, 'identityKey' | 'nonce' | 'timestamp' | 'body'>): Uint8Array {
  const ts = new Uint8Array(8);
  new DataView(ts.buffer).setBigUint64(0, BigInt(env.timestamp), true);
  const buf = new Uint8Array(env.identityKey.length + env.nonce.length + 8 + env.body.length);
  let off = 0;
  buf.set(env.identityKey, off);
  off += env.identityKey.length;
  buf.set(env.nonce, off);
  off += env.nonce.length;
  buf.set(ts, off);
  off += 8;
  buf.set(env.body, off);
  return nobleSha256(buf);
}

/** Build + sign an envelope. Returns the parts in a postMessage-friendly shape. */
export function buildEnvelope(
  identitySk: Uint8Array,
  identityPk: Uint8Array,
  body: Uint8Array,
  opts?: { now?: () => number; nonce?: Uint8Array },
): {
  identityKey: string;
  nonce: string;
  timestamp: number;
  signature: string;
  body: Uint8Array;
} {
  if (identitySk.length !== 32) throw new Error('identity sk must be 32 bytes');
  if (identityPk.length !== 33) throw new Error('identity pk must be 33-byte compressed');
  const nonce = opts?.nonce ?? crypto.getRandomValues(new Uint8Array(32));
  if (nonce.length !== 32) throw new Error('nonce must be 32 bytes');
  const timestamp = (opts?.now ?? (() => Math.floor(Date.now() / 1000)))() >>> 0;
  const digest = envelopeDigest({ identityKey: identityPk, nonce, timestamp, body });
  const sig = secp.sign(digest, identitySk).normalizeS();
  const der = encodeDer(sig.r, sig.s);
  return {
    identityKey: bytesToHex(identityPk),
    nonce: bytesToHex(nonce),
    timestamp,
    signature: bytesToHex(der),
    body,
  };
}

/**
 * Parse an envelope-shaped object (as delivered via postMessage) and verify
 * the signature + replay window. Returns either the parsed envelope or a
 * structured rejection reason — the caller decides how to surface this to
 * the requesting dApp.
 */
export function parseEnvelope(
  raw: unknown,
  opts?: { now?: () => number; replayWindowSecs?: number },
): Brc100ParseResult {
  if (typeof raw !== 'object' || raw === null) {
    return { ok: false, reason: 'envelope: not an object' };
  }
  const r = raw as Record<string, unknown>;
  const requiredStringKeys = ['identityKey', 'nonce', 'signature'] as const;
  for (const k of requiredStringKeys) {
    if (typeof r[k] !== 'string') return { ok: false, reason: `envelope: missing ${k}` };
  }
  if (typeof r.timestamp !== 'number' && typeof r.timestamp !== 'string') {
    return { ok: false, reason: 'envelope: missing timestamp' };
  }
  if (!(r.body instanceof Uint8Array) && typeof r.body !== 'string') {
    return { ok: false, reason: 'envelope: missing body' };
  }

  let identityKey: Uint8Array, nonce: Uint8Array, signature: Uint8Array;
  try {
    identityKey = hexToBytes(r.identityKey as string);
    nonce = hexToBytes(r.nonce as string);
    signature = hexToBytes(r.signature as string);
  } catch (e) {
    return { ok: false, reason: `envelope: hex decode — ${(e as Error).message}` };
  }
  if (identityKey.length !== 33) return { ok: false, reason: 'envelope: identityKey must be 33 bytes' };
  if (nonce.length !== 32) return { ok: false, reason: 'envelope: nonce must be 32 bytes' };

  const timestamp = typeof r.timestamp === 'string' ? parseInt(r.timestamp, 10) : (r.timestamp as number);
  if (!Number.isFinite(timestamp) || timestamp < 0) {
    return { ok: false, reason: 'envelope: bad timestamp' };
  }

  const replayWindow = opts?.replayWindowSecs ?? DEFAULT_REPLAY_WINDOW_SECS;
  const now = (opts?.now ?? (() => Math.floor(Date.now() / 1000)))();
  if (Math.abs(now - timestamp) > replayWindow) {
    return { ok: false, reason: `envelope: timestamp outside ${replayWindow}s window` };
  }

  const body = r.body instanceof Uint8Array ? r.body : new TextEncoder().encode(r.body as string);

  // Verify ECDSA signature over the canonical digest.
  const digest = envelopeDigest({ identityKey, nonce, timestamp, body });
  let parsed: { r: bigint; s: bigint };
  try {
    parsed = decodeDer(signature);
  } catch (e) {
    return { ok: false, reason: `envelope: signature DER — ${(e as Error).message}` };
  }
  const sig = new secp.Signature(parsed.r, parsed.s);
  let ok = false;
  try {
    ok = secp.verify(sig, digest, identityKey, { lowS: false });
  } catch (e) {
    return { ok: false, reason: `envelope: verify — ${(e as Error).message}` };
  }
  if (!ok) return { ok: false, reason: 'envelope: signature failed verification' };

  return { ok: true, envelope: { identityKey, nonce, timestamp, signature, body } };
}

export { hexToBytes, bytesToHex };
