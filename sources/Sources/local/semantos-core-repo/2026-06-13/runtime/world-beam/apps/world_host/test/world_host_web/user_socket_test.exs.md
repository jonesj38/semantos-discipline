defmodule WorldHostWeb.UserSocketTest do
  @moduledoc """
  D-V3 / D-A1 — `WorldHostWeb.UserSocket.connect/3` contract.

  These tests pin the integration boundary D-V3 introduced and D-A1
  hardened: every inbound WebSocket presents a BRC-100 SignedBundle
  wrapping a BRC-52 cert; connect/3 dispatches to the Verifier Sidecar
  (via the `WorldHost.VerifierClient` behaviour); accepted sockets
  carry `socket.assigns.bca`, `.cert_id`, and `.identity_key`; rejected
  sockets surface a structured error tuple.

  D-A1: the test fixtures are now **really signed** ECDSA bundles
  generated by `WorldHost.Test.SignedBundleFixture` (the test owns the
  keypair). The mock (`WorldHost.VerifierClient.Mock`) verifies them
  locally with the same Phase 1 + Phase 2 logic the production
  `BrcVerifier` runs.

  Spec source: `docs/spec/protocol-v0.5.md` §4.2 (BRC-52 cert),
               §9.5 (Verifier Sidecar),
               §12.1 (SignedBundle envelope).
  Canonical terms: SignedBundle, Verifier Sidecar, BCA, cert_id.
  K invariant: K2 — boundary verification before state mutation.
  """

  use WorldHostWeb.ChannelCase, async: false

  alias WorldHostWeb.UserSocket

  setup do
    Process.put(:verifier_mock_mode, :verify)
    Process.delete({:verifier_mock, :error})
    :ok
  end

  describe "connect/3 — rejects without a signed_bundle param" do
    test "missing signed_bundle returns {:error, %{code: \"envelope_missing\"}}" do
      assert {:error, %{code: "envelope_missing"} = err} =
               connect(UserSocket, %{})

      assert err.message =~ "signed_bundle"
    end

    test "rejected connect leaves no bca on the socket" do
      # connect/3 returns {:error, _} — the caller never gets a socket
      # to read assigns from. Pinning the negative case explicitly so
      # a future regression that leaks a half-initialised socket is caught.
      assert {:error, _} = connect(UserSocket, %{})
    end

    test "non-JSON signed_bundle returns envelope_malformed" do
      assert {:error, %{code: "envelope_malformed"}} =
               connect(UserSocket, %{"signed_bundle" => "{ not json"})
    end
  end

  describe "connect/3 — rejects on bad signature" do
    test "verifier returns ok:false → connect/3 returns {:error, %{code: ...}}" do
      Process.put(:verifier_mock_mode, :reject)

      Process.put({:verifier_mock, :error}, %{
        ok: false,
        code: "brc100_invalid_signature",
        message: "signature verification failed"
      })

      assert {:error, %{code: "brc100_invalid_signature", message: msg}} =
               connect(UserSocket, mock_connect_params("rejected-seed"))

      assert msg =~ "signature verification failed"
    end

    test "verifier returns identity-binding mismatch → connect/3 propagates the code" do
      Process.put(:verifier_mock_mode, :reject)

      Process.put({:verifier_mock, :error}, %{
        ok: false,
        code: "brc52_identity_binding_mismatch",
        message: "K2 binding check failed"
      })

      assert {:error, %{code: "brc52_identity_binding_mismatch"}} =
               connect(UserSocket, mock_connect_params("seed-2"))
    end

    test "tampered signature in a real bundle is rejected by local verification" do
      # D-A1: real verification path. Mint a real bundle, then flip the
      # first signature byte and confirm the mock rejects the bundle
      # with `brc100_invalid_signature` — same code the production
      # sidecar would return.
      params = mock_connect_params("alice-seed-tampered")
      bundle = params["signed_bundle"]

      tampered_sig =
        case bundle["x-brc100-signature"] do
          <<"aa", rest::binary>> -> "bb" <> rest
          <<_a, _b, rest::binary>> -> "aa" <> rest
        end

      tampered = Map.put(bundle, "x-brc100-signature", tampered_sig)

      assert {:error, %{code: "brc100_invalid_signature"}} =
               connect(UserSocket, %{"signed_bundle" => tampered})
    end
  end

  describe "connect/3 — accepts a verified envelope" do
    test "valid bundle → socket.assigns.bca + .cert_id set; identity_key carried" do
      %{bundle: bundle, cert: cert} = build_fixture("alice-seed")
      assert {:ok, socket} = connect(UserSocket, %{"signed_bundle" => bundle})

      assert socket.assigns.cert_id == cert["certId"]
      assert is_binary(socket.assigns.bca)
      assert byte_size(socket.assigns.bca) == 32
      assert socket.assigns.identity_key == bundle["x-brc100-identitykey"]
    end

    test "same seed → same cert_id + bca across two connect/3 calls (deterministic)" do
      params1 = mock_connect_params("bob-seed")
      params2 = mock_connect_params("bob-seed")

      assert {:ok, s1} = connect(UserSocket, params1)
      assert {:ok, s2} = connect(UserSocket, params2)

      assert s1.assigns.cert_id == s2.assigns.cert_id
      assert s1.assigns.bca == s2.assigns.bca
    end

    test "different seeds → different cert_ids and different bcas" do
      assert {:ok, s1} = connect(UserSocket, mock_connect_params("seed-A"))
      assert {:ok, s2} = connect(UserSocket, mock_connect_params("seed-B"))

      refute s1.assigns.cert_id == s2.assigns.cert_id
      refute s1.assigns.bca == s2.assigns.bca
    end

    test "id/1 returns a stable socket id derived from cert_id" do
      assert {:ok, socket} = connect(UserSocket, mock_connect_params("carol-seed"))
      assert UserSocket.id(socket) == "socket:" <> socket.assigns.cert_id
    end
  end

  describe "connect/3 — cap_token forwarding" do
    test "cap_token param is forwarded to the verifier" do
      # Force the mock to capture what it received via the process dict.
      defmodule CaptureMock do
        @behaviour WorldHost.VerifierClient

        @impl WorldHost.VerifierClient
        def verify(envelope, cap_token) do
          send(self(), {:capture_mock, envelope, cap_token})
          %{ok: true, cert_id: "captured-cert", bca: "00" |> String.duplicate(16)}
        end
      end

      prior = Application.get_env(:world_host, :verifier_client)
      Application.put_env(:world_host, :verifier_client, CaptureMock)

      try do
        params =
          mock_connect_params("seed-cap")
          |> Map.put("cap_token", %{"txId" => String.duplicate("a", 64), "vout" => 0})

        assert {:ok, _} = connect(UserSocket, params)
        assert_received {:capture_mock, envelope, cap_token}
        assert is_map(envelope)
        assert cap_token["txId"] == String.duplicate("a", 64)
        assert cap_token["vout"] == 0
      after
        if prior, do: Application.put_env(:world_host, :verifier_client, prior)
      end
    end
  end
end
