// Phase W6 — BRC-100 envelope parse / build + signature verification.
//
// Mirrors the W5 TS bridge (`cartridges/wallet-headers/brain/src/brc100.ts`) so the
// sovereign-node daemon and the browser wallet bundle interop on a single
// canonical signed-payload digest.
//
// CANONICAL DIGEST FORMULA (per docs/design/BRC100-CANONICAL-DIGEST.md):
//
//     digest = SHA256( identityKey(33) ‖ nonce(32) ‖ ts_le8 ‖ body )
//
// The wire format for this transport (WSS):
//
//   {
//     "headers": {
//       "x-brc100-identitykey": "<33-byte sec1, hex>",
//       "x-brc100-nonce":       "<32-byte hex>",
//       "x-brc100-timestamp":   "<u32 unix seconds, decimal string>",
//       "x-brc100-signature":   "<DER bytes, hex>"
//     },
//     "body": "<utf-8 string — opaque to envelope; for RPC this is
//              JSON.stringify({method, params, id})>"
//   }
//
// The signature commits to the digest under the wallet's identity private
// key. The daemon parses `body` as JSON internally to route RPC, but the
// digest is computed over body bytes as they appear on the wire.
//
// ─── Method coverage (v0.1) ───────────────────────────────────────────
//
// W6 acceptance criterion 3 names `getPublicKey` as the smoke-test method.
// Other BRC-100 methods (`createAction`, `signAction`, …) return a structured
// 501 reject so dApps see "not implemented" rather than a hang.
//
// Failure-atomicity: any malformed-envelope path returns a structured
// `Reject` without mutating state.

const std = @import("std");
const host = @import("host");

pub const ParseError = error{
    InvalidJson,
    MissingHeaders,
    MissingBody,
    InvalidIdentityKey,
    InvalidSignature,
    InvalidNonce,
    InvalidTimestamp,
    BodyNotString,
    OutOfMemory,
};

pub const VerifyError = error{
    BadDigest,
    BadSignature,
};

pub const DispatchError = error{
    NotImplemented,
    InternalError,
    OutOfMemory,
};

/// Parsed view into a BRC-100 envelope. All slices borrow into the
/// caller-owned JSON buffer — they're invalid after the buffer is freed.
pub const Envelope = struct {
    /// 33-byte SEC1-compressed identity public key.
    identity_key: [33]u8,
    /// 32-byte nonce.
    nonce: [32]u8,
    /// Unix seconds — the on-the-wire timestamp.
    timestamp_secs: u64,
    /// DER ECDSA signature bytes (no sighash byte appended).
    signature: []const u8,
    /// Opaque body bytes that the signature commits to. For RPC use this
    /// is a UTF-8 JSON string of `{method, params, id}`; the dispatcher
    /// parses it lazily via `parseRpc`.
    body: []const u8,
};

pub const RpcRequest = struct {
    method: []const u8,
    params_json: []const u8,
    id: []const u8,
};

pub const Reject = struct {
    code: i32,
    message: []const u8,
};

/// Compute the canonical signed-payload digest. Single source of truth for
/// the formula in `docs/design/BRC100-CANONICAL-DIGEST.md` §1.
pub fn computeDigest(
    identity_key: *const [33]u8,
    nonce: *const [32]u8,
    timestamp_secs: u64,
    body: []const u8,
    out: *[32]u8,
) void {
    // Stack-build the digest input: 33 + 32 + 8 + body_len bytes.
    // For typical RPC bodies (< 1 KB) this fits comfortably in a stack
    // buffer; for larger bodies we fall back to a heap alloc via the
    // streaming Sha256 API.
    const fixed_prefix_len: usize = 33 + 32 + 8;
    if (body.len <= 4096 - fixed_prefix_len) {
        var buf: [4096]u8 = undefined;
        @memcpy(buf[0..33], identity_key);
        @memcpy(buf[33..65], nonce);
        std.mem.writeInt(u64, buf[65..73], timestamp_secs, .little);
        @memcpy(buf[73 .. 73 + body.len], body);
        host.sha256(buf[0 .. 73 + body.len], out);
    } else {
        // Streaming path — body is large.
        var hasher = std.crypto.hash.sha2.Sha256.init(.{});
        hasher.update(identity_key);
        hasher.update(nonce);
        var ts_buf: [8]u8 = undefined;
        std.mem.writeInt(u64, &ts_buf, timestamp_secs, .little);
        hasher.update(&ts_buf);
        hasher.update(body);
        hasher.final(out);
    }
}

/// Parse + verify a BRC-100 envelope. The returned envelope's slices borrow
/// into heap-duped buffers owned by the caller; release via `freeEnvelope`.
pub fn parse(
    allocator: std.mem.Allocator,
    raw: []const u8,
) ParseError!Envelope {
    var parsed = std.json.parseFromSlice(std.json.Value, allocator, raw, .{}) catch return error.InvalidJson;
    defer parsed.deinit();

    const root = parsed.value;
    if (root != .object) return error.InvalidJson;

    const headers_val = root.object.get("headers") orelse return error.MissingHeaders;
    if (headers_val != .object) return error.MissingHeaders;

    const ik_val = headers_val.object.get("x-brc100-identitykey") orelse return error.MissingHeaders;
    const nonce_val = headers_val.object.get("x-brc100-nonce") orelse return error.MissingHeaders;
    const ts_val = headers_val.object.get("x-brc100-timestamp") orelse return error.MissingHeaders;
    const sig_val = headers_val.object.get("x-brc100-signature") orelse return error.MissingHeaders;

    if (ik_val != .string or nonce_val != .string or ts_val != .string or sig_val != .string) {
        return error.MissingHeaders;
    }

    var ik_bytes: [33]u8 = undefined;
    decodeHexExact(33, ik_val.string, &ik_bytes) catch return error.InvalidIdentityKey;

    var nonce_bytes: [32]u8 = undefined;
    decodeHexExact(32, nonce_val.string, &nonce_bytes) catch return error.InvalidNonce;

    const ts = std.fmt.parseInt(u64, ts_val.string, 10) catch return error.InvalidTimestamp;

    // Signature heap-duped because the JSON tree is freed below.
    if (sig_val.string.len % 2 != 0 or sig_val.string.len < 2) return error.InvalidSignature;
    const sig_buf = allocator.alloc(u8, sig_val.string.len / 2) catch return error.OutOfMemory;
    errdefer allocator.free(sig_buf);
    decodeHexInto(sig_val.string, sig_buf) catch return error.InvalidSignature;

    const body_val = root.object.get("body") orelse return error.MissingBody;
    if (body_val != .string) return error.BodyNotString;

    const body_dup = allocator.dupe(u8, body_val.string) catch return error.OutOfMemory;
    errdefer allocator.free(body_dup);

    return .{
        .identity_key = ik_bytes,
        .nonce = nonce_bytes,
        .timestamp_secs = ts,
        .signature = sig_buf,
        .body = body_dup,
    };
}

/// Free heap allocations attached to a parsed envelope.
pub fn freeEnvelope(allocator: std.mem.Allocator, env: *const Envelope) void {
    allocator.free(env.signature);
    allocator.free(env.body);
}

/// Verify the BRC-100 transport signature against the parsed envelope.
/// Uses `host.checksig` which in the full profile dispatches to
/// `bsvz.crypto.verifyDigest256RelaxedSec1`. The transport signature has
/// NO sighash byte, so we append a single 0x00 to satisfy `host.checksig`'s
/// contract (it strips the last byte before DER-parsing).
pub fn verify(env: *const Envelope, allocator: std.mem.Allocator) VerifyError!void {
    var digest: [32]u8 = undefined;
    computeDigest(&env.identity_key, &env.nonce, env.timestamp_secs, env.body, &digest);

    const sig_with_sighash = allocator.alloc(u8, env.signature.len + 1) catch return error.BadSignature;
    defer allocator.free(sig_with_sighash);
    @memcpy(sig_with_sighash[0..env.signature.len], env.signature);
    sig_with_sighash[env.signature.len] = 0x00;

    const ok = host.checksig(&env.identity_key, &digest, sig_with_sighash);
    if (!ok) return error.BadSignature;
}

/// Parse the envelope's body as a JSON-RPC request. Returns a borrowed view
/// — slices point into the envelope's body buffer, valid until
/// `freeEnvelope` runs.
pub fn parseRpc(
    allocator: std.mem.Allocator,
    env: *const Envelope,
    out_id: *[]u8,
) ParseError!RpcRequest {
    var parsed = std.json.parseFromSlice(std.json.Value, allocator, env.body, .{}) catch return error.InvalidJson;
    defer parsed.deinit();
    if (parsed.value != .object) return error.InvalidJson;

    const method_val = parsed.value.object.get("method") orelse return error.InvalidJson;
    const id_val = parsed.value.object.get("id") orelse return error.InvalidJson;
    if (method_val != .string or id_val != .string) return error.InvalidJson;

    const params_json = extractParamsSlice(env.body) orelse return error.InvalidJson;

    // Heap-dupe id so it survives the JSON tree's deinit. Caller owns it.
    out_id.* = allocator.dupe(u8, id_val.string) catch return error.OutOfMemory;
    errdefer allocator.free(out_id.*);

    // method and params_json borrow from env.body — no dupe needed.
    const method_in_body = std.mem.indexOf(u8, env.body, method_val.string) orelse return error.InvalidJson;
    return .{
        .method = env.body[method_in_body .. method_in_body + method_val.string.len],
        .params_json = params_json,
        .id = out_id.*,
    };
}

// ─── Response builder ────────────────────────────────────────────────

/// Build a JSON response envelope. Caller owns the returned bytes.
pub fn buildResponse(
    allocator: std.mem.Allocator,
    request_id: []const u8,
    /// JSON object literal, e.g. `{"publicKey":"02ab…"}`. Inserted verbatim.
    result_json: []const u8,
) ![]u8 {
    return try std.fmt.allocPrint(
        allocator,
        "{{\"id\":\"{s}\",\"result\":{s}}}",
        .{ request_id, result_json },
    );
}

pub fn buildReject(
    allocator: std.mem.Allocator,
    request_id: []const u8,
    code: i32,
    message: []const u8,
) ![]u8 {
    return try std.fmt.allocPrint(
        allocator,
        "{{\"id\":\"{s}\",\"error\":{{\"code\":{d},\"message\":\"{s}\"}}}}",
        .{ request_id, code, message },
    );
}

// ─── Helpers ─────────────────────────────────────────────────────────

fn decodeHexExact(comptime N: usize, hex: []const u8, out: *[N]u8) !void {
    if (hex.len != N * 2) return error.InvalidHex;
    _ = try std.fmt.hexToBytes(out, hex);
}

fn decodeHexInto(hex: []const u8, out: []u8) !void {
    if (hex.len != out.len * 2) return error.InvalidHex;
    _ = try std.fmt.hexToBytes(out, hex);
}

/// Locate the JSON `params` field's verbatim substring within a JSON object
/// — returns the slice from `{` to the matching `}`. Used by `parseRpc`.
fn extractParamsSlice(raw: []const u8) ?[]const u8 {
    const needle = "\"params\"";
    const start = std.mem.indexOf(u8, raw, needle) orelse return null;
    var i: usize = start + needle.len;
    while (i < raw.len and (raw[i] == ' ' or raw[i] == '\t' or raw[i] == '\n' or raw[i] == '\r')) i += 1;
    if (i >= raw.len or raw[i] != ':') return null;
    i += 1;
    while (i < raw.len and (raw[i] == ' ' or raw[i] == '\t' or raw[i] == '\n' or raw[i] == '\r')) i += 1;
    if (i >= raw.len or raw[i] != '{') return null;

    const obj_start = i;
    var depth: u32 = 0;
    var in_string = false;
    var escape = false;
    while (i < raw.len) : (i += 1) {
        const c = raw[i];
        if (escape) {
            escape = false;
            continue;
        }
        if (in_string) {
            if (c == '\\') {
                escape = true;
            } else if (c == '"') {
                in_string = false;
            }
            continue;
        }
        if (c == '"') {
            in_string = true;
            continue;
        }
        if (c == '{') depth += 1;
        if (c == '}') {
            depth -= 1;
            if (depth == 0) return raw[obj_start .. i + 1];
        }
    }
    return null;
}

// ─── Tests ────────────────────────────────────────────────────────────

test "computeDigest is deterministic" {
    const ik = [_]u8{0x02} ++ [_]u8{0x11} ** 32;
    const nonce = [_]u8{0xAB} ** 32;
    const body = "hello";
    var d1: [32]u8 = undefined;
    var d2: [32]u8 = undefined;
    computeDigest(&ik, &nonce, 1234567890, body, &d1);
    computeDigest(&ik, &nonce, 1234567890, body, &d2);
    try std.testing.expectEqualSlices(u8, &d1, &d2);
}

test "computeDigest changes when body changes" {
    const ik = [_]u8{0x02} ++ [_]u8{0x11} ** 32;
    const nonce = [_]u8{0xAB} ** 32;
    var d1: [32]u8 = undefined;
    var d2: [32]u8 = undefined;
    computeDigest(&ik, &nonce, 1234567890, "a", &d1);
    computeDigest(&ik, &nonce, 1234567890, "b", &d2);
    try std.testing.expect(!std.mem.eql(u8, &d1, &d2));
}

test "extractParamsSlice handles nested objects" {
    const env = "{\"method\":\"x\",\"params\":{\"a\":1,\"b\":{\"c\":2}},\"id\":\"q\"}";
    const ps = extractParamsSlice(env).?;
    try std.testing.expectEqualSlices(u8, "{\"a\":1,\"b\":{\"c\":2}}", ps);
}

test "extractParamsSlice handles strings with braces" {
    const env = "{\"params\":{\"s\":\"a{b}c\"},\"id\":\"q\"}";
    const ps = extractParamsSlice(env).?;
    try std.testing.expectEqualSlices(u8, "{\"s\":\"a{b}c\"}", ps);
}

test "buildResponse round-trips id" {
    const out = try buildResponse(std.testing.allocator, "req-1", "{\"publicKey\":\"02ab\"}");
    defer std.testing.allocator.free(out);
    try std.testing.expectEqualSlices(u8, "{\"id\":\"req-1\",\"result\":{\"publicKey\":\"02ab\"}}", out);
}
