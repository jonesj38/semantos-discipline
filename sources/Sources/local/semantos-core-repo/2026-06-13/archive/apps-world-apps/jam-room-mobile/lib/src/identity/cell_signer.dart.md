// D-O5m.followup-8 capture+upload — Pure-Dart ECDSA-secp256k1-sha256
// cell signer.
//
// Reference: runtime/semantos-brain/src/signed_bundle.zig::verifySignature (the
//            canonical brain-side verifier scheme; recovery-byte loop
//            over 31..34); runtime/semantos-brain/src/bkds.zig (secp256k1 +
//            ECDSA primitives via bsvz the brain side uses). The
//            fixture at test/fixtures/cell-signing-fixture.json is
//            generated by Zig (tests/cell_signing_fixture_emit.zig)
//            and asserts byte-for-byte parity for the {priv, payload,
//            expectedSignature} tuple — without that the brain might
//            accept Dart-signed cells that brain-signed ones would reject
//            or vice versa.
//
// Algorithm:
//   1. digest = SHA-256(payloadBytes)
//   2. (r, s) = ECDSA_sign(secp256k1, devicePriv, digest) with
//      RFC 6979 deterministic-k (no noise) — pointycastle's
//      ECDSASigner with Mac("SHA-256/HMAC/SHA-256")
//   3. Normalize s to low-s — `s = min(s, n - s)` where `n` is the
//      secp256k1 curve order
//   4. Return 32-byte big-endian r || 32-byte big-endian s (64 bytes
//      compact, no recovery byte)
//
// Why no recovery byte:
//   The brain verifier (`signed_bundle.zig::verifySignature`) tries all
//   four recovery bytes (31..34) regardless of what the signer
//   stamped. The cell-signing protocol therefore only carries the
//   (r,s) compact signature; the brain recovers the pubkey and matches
//   it against the device's child cert pubkey. Skipping the recovery
//   byte saves a byte and keeps the wire shape minimal.
//
// What `verifyCellSignature` does:
//   The same reverse-engineered pubkey-recovery loop the brain uses,
//   exposed as a Dart helper for the unit-test parity assertions.
//   Production-side mobile code does NOT call verify — the brain is
//   the verifier; the device only signs.
//
// Why pointycastle (not flutter_secure_storage's ECC handle):
//   D-O5m.followup-2 lands the Keychain/Keystore-backed signing key
//   handle migration.  In the new path, the priv lives in iOS
//   Keychain / Android EncryptedSharedPreferences (with biometric
//   gating + AndroidKeyStore-master-key wrapping at rest), and the
//   priv enters Dart memory only briefly during a sign call.  The
//   pure-Dart pointycastle implementation below stays in place as
//   the InMemoryAdapter fallback + the canonical wire-byte
//   reference.  The new `CellSigner` class wraps a
//   `SecureSigningKeyAdapter` so existing call-sites can opt into
//   the secure path without changing their public surface; the
//   bare `signCellPayload(payload, priv)` function stays available
//   for legacy raw-priv records that haven't migrated yet.
//
// Honest scope (mirrored in the runbook):
//   secp256k1 is NOT supported by iOS Secure Enclave (NIST P-256
//   only) or AndroidKeyStore EC keys (NIST curves only), so the
//   priv DOES briefly enter process memory during sign operations.
//   The migration adds at-rest encryption + biometric gating + key
//   revocation but is NOT a true "key never leaves enclave"
//   implementation — that's a future revision requiring either a
//   curve change or a JNI/CMake-built secp256k1 lib running in a
//   secure context.

import 'dart:typed_data';

import 'package:pointycastle/digests/sha256.dart';
import 'package:pointycastle/ecc/api.dart';
import 'package:pointycastle/ecc/curves/secp256k1.dart';

import 'secure_signing_key.dart';

/// Length of a compact (r || s) cell signature in bytes.
const int cellSignatureLength = 64;

/// secp256k1 domain parameters — same instance brc42_derive.dart uses.
final ECDomainParameters _secp256k1 = ECCurve_secp256k1();

/// Zig-stdlib-compatible deterministic-k length for secp256k1+sha256.
/// Zig stdlib's `deterministicScalar` always reserves 32 bytes for the
/// noise slot regardless of whether noise was supplied — when `noise`
/// is null, those bytes are zeros.  bsvz's `signCompact` passes
/// `noise=null` (see Zig stdlib `EcdsaSha256d.KeyPair.signPrehashed`
/// line 359-361 calling `Signer.init(secret_key, noise)` with the
/// pass-through noise).  So our wire-compatible RFC 6979 layout is:
///
///   V (32)  ||  0x00  ||  zeros (32)  ||  secret_key (32)  ||  h (32)
///
/// where `V` initialises to all-0x01 and `h` is the 32-byte SHA-256
/// digest.  This deviates from the canonical RFC 6979 layout (no
/// extra-zero noise slot) but matches Zig stdlib byte-for-byte —
/// load-bearing for the `cell-signing-fixture.json` parity proof.
const int _zigNoiseLength = 32;

/// Sign `payloadBytes` with `devicePrivBytes` using a Zig-stdlib-
/// compatible deterministic-k ECDSA over SHA-256, normalising the
/// result to low-s. Returns 64 raw bytes (r big-endian || s big-
/// endian).
Uint8List signCellPayload(Uint8List payloadBytes, Uint8List devicePrivBytes) {
  if (devicePrivBytes.length != 32) {
    throw ArgumentError(
        'devicePrivBytes must be 32 bytes, got ${devicePrivBytes.length}');
  }

  // Compute SHA-256(payload) — the digest is what RFC 6979's k
  // derivation consumes AND what enters the (e + r*x) * k^-1 formula.
  final sha = SHA256Digest();
  final digest = Uint8List(32);
  sha.update(payloadBytes, 0, payloadBytes.length);
  sha.doFinal(digest, 0);

  final n = _secp256k1.n;
  final d = _bytesToBigInt(devicePrivBytes);

  // Zig-stdlib-compatible deterministic-k.  See `_zigNoiseLength`
  // header for the layout rationale.
  BigInt? k;
  BigInt? r;
  BigInt? s;
  final kIter = _ZigCompatibleKCalculator(devicePrivBytes, digest);
  for (var attempt = 0; attempt < 1024; attempt++) {
    k = kIter.nextK();
    final p = (_secp256k1.G * k)!;
    final x = p.x!.toBigInteger()!;
    r = x % n;
    if (r == BigInt.zero) continue;

    // e = bits2int(digest) — for sha256+secp256k1, the bit lengths
    // match (256 each), so e is just the digest as a big-endian
    // integer.  No reduction here (the modular arithmetic at the
    // formula end handles the >n case).
    final e = _bytesToBigInt(digest);

    final kInv = k.modInverse(n);
    s = (kInv * (e + (d * r))) % n;
    if (s != BigInt.zero) break;
  }
  if (k == null || r == null || s == null) {
    throw StateError('signing failed: deterministic-k loop exhausted');
  }

  final halfN = n >> 1;
  // Low-s normalisation: if s > n/2 then s = n - s. Mirrors BIP-62 and
  // the bsvz/Zig recovery semantic — the brain's recovery loop
  // accepts both branches but the spec pins low-s to keep the wire
  // bytes deterministic.
  final lowS = s.compareTo(halfN) > 0 ? n - s : s;

  final out = Uint8List(cellSignatureLength);
  _writeBigIntBE(r, out, 0, 32);
  _writeBigIntBE(lowS, out, 32, 32);
  return out;
}

/// SHA-256 of SHA-256 — the Zig-stdlib `EcdsaSecp256k1Sha256oSha256`
/// scheme uses this as the HMAC's underlying hash (a length-extension
/// defense; not strictly needed for SHA-256 but built into the Zig
/// stdlib's bsvz-compatible signing scheme).  Implementation:
/// `Sha256oSha256.hash(b) = Sha256(Sha256(b))`.
Uint8List _sha256oSha256(Uint8List bytes) {
  final inner = SHA256Digest();
  final innerOut = Uint8List(32);
  inner.update(bytes, 0, bytes.length);
  inner.doFinal(innerOut, 0);
  final outer = SHA256Digest();
  final outerOut = Uint8List(32);
  outer.update(innerOut, 0, innerOut.length);
  outer.doFinal(outerOut, 0);
  return outerOut;
}

/// HMAC-Sha256oSha256 — manually composed because pointycastle doesn't
/// ship a Sha256oSha256 digest.  The construction matches RFC 2104:
///
///   HMAC(K, M) = H((K' xor opad) || H((K' xor ipad) || M))
///
/// where K' = K if |K| <= block_length else H(K), padded with zeros to
/// block_length.  For Sha256oSha256 the block_length is 64 (matches
/// inner sha256) and the digest_length is 32.  Zig's stdlib
/// `Hmac(Sha256oSha256)` uses these same constants, so the byte
/// outputs agree.
Uint8List _hmacSha256oSha256(Uint8List key, Uint8List msg) {
  const blockLen = 64;
  Uint8List k0;
  if (key.length > blockLen) {
    k0 = _sha256oSha256(key);
    final padded = Uint8List(blockLen);
    padded.setRange(0, k0.length, k0);
    k0 = padded;
  } else if (key.length < blockLen) {
    k0 = Uint8List(blockLen);
    k0.setRange(0, key.length, key);
  } else {
    k0 = Uint8List.fromList(key);
  }

  final ipad = Uint8List(blockLen);
  final opad = Uint8List(blockLen);
  for (var i = 0; i < blockLen; i++) {
    ipad[i] = k0[i] ^ 0x36;
    opad[i] = k0[i] ^ 0x5c;
  }

  final inner = Uint8List(blockLen + msg.length);
  inner.setRange(0, blockLen, ipad);
  inner.setRange(blockLen, inner.length, msg);
  final innerHash = _sha256oSha256(inner);

  final outer = Uint8List(blockLen + innerHash.length);
  outer.setRange(0, blockLen, opad);
  outer.setRange(blockLen, outer.length, innerHash);
  return _sha256oSha256(outer);
}

/// Zig-stdlib-compatible RFC 6979-style deterministic-k iterator.
/// Matches `deterministicScalar` in `std/crypto/ecdsa.zig` byte-for-
/// byte for the secp256k1+Sha256oSha256 scheme that bsvz's
/// `signCompact` rides on top of.  Two key deviations from the
/// canonical RFC 6979:
///   1. The HMAC underlying hash is Sha256(Sha256(.)) (length-
///      extension defense), not plain SHA-256.
///   2. A 32-byte zero "noise" slot sits between the V/0x00 prefix
///      and the secret key (Zig stdlib reserves the slot regardless
///      of whether real noise was supplied).
class _ZigCompatibleKCalculator {
  late Uint8List _v;
  late Uint8List _k;
  final Uint8List _x;
  final Uint8List _h;
  static final BigInt _n = ECCurve_secp256k1().n;

  _ZigCompatibleKCalculator(Uint8List secretKey, Uint8List digest)
      : _x = secretKey,
        _h = digest {
    _v = Uint8List(32)..fillRange(0, 32, 0x01);
    _k = Uint8List(32);

    // Step (d): K = HMAC(K, V || 0x00 || zeros(32) || x || h)
    final m = _buildM(0x00);
    _k = _hmacSha256oSha256(_k, m);
    // V = HMAC(K, V)
    _v = _hmacSha256oSha256(_k, _v);
    // K = HMAC(K, V || 0x01 || zeros(32) || x || h)
    final m2 = _buildM(0x01);
    _k = _hmacSha256oSha256(_k, m2);
    // V = HMAC(K, V)
    _v = _hmacSha256oSha256(_k, _v);
  }

  Uint8List _buildM(int octet) {
    // V (32) || octet (1) || zeros(32) || x (32) || h (32)
    final m = Uint8List(32 + 1 + _zigNoiseLength + 32 + 32);
    m.setRange(0, 32, _v);
    m[32] = octet;
    // zeros(32) — already zero from default Uint8List
    m.setRange(32 + 1 + _zigNoiseLength, 32 + 1 + _zigNoiseLength + 32, _x);
    m.setRange(32 + 1 + _zigNoiseLength + 32, m.length, _h);
    return m;
  }

  BigInt nextK() {
    while (true) {
      _v = _hmacSha256oSha256(_k, _v);
      final candidate = _bytesToBigInt(_v);
      if (candidate != BigInt.zero && candidate < _n) {
        return candidate;
      }
      // Reseed: K = HMAC(K, V || 0x00); V = HMAC(K, V).
      final reseed = Uint8List(32 + 1)
        ..setRange(0, 32, _v)
        ..[32] = 0x00;
      _k = _hmacSha256oSha256(_k, reseed);
      _v = _hmacSha256oSha256(_k, _v);
    }
  }
}

/// Verify a 64-byte (r || s) compact signature against a 33-byte
/// compressed-SEC1 device pubkey by recovering the pubkey from the
/// signature + payload digest. Returns true on a match. Mirrors the
/// brain-side verifier (`runtime/semantos-brain/src/signed_bundle.zig::
/// verifySignature` — recovery-byte loop over 31..34) so the test
/// suite can assert correctness without round-tripping through the
/// brain.
bool verifyCellSignature(
    Uint8List payloadBytes, Uint8List signature, Uint8List devicePubBytes) {
  if (signature.length != cellSignatureLength) return false;
  if (devicePubBytes.length != 33) return false;

  final digestBytes = Uint8List(32);
  final sha = SHA256Digest();
  sha.update(payloadBytes, 0, payloadBytes.length);
  sha.doFinal(digestBytes, 0);

  final r = _bytesToBigInt(signature.sublist(0, 32));
  final s = _bytesToBigInt(signature.sublist(32, 64));
  if (r == BigInt.zero || s == BigInt.zero) return false;

  final expectedPub = _secp256k1.curve.decodePoint(devicePubBytes);
  if (expectedPub == null) return false;

  // Pubkey-recovery loop: try recId 0..3, accept on a match.
  for (var recId = 0; recId < 4; recId++) {
    final candidate = _recoverPubkey(r, s, digestBytes, recId);
    if (candidate == null) continue;
    final candCompressed = candidate.getEncoded(true);
    if (_bytesEqual(candCompressed, devicePubBytes)) return true;
  }
  return false;
}

// ─── Pubkey recovery (SEC1 4.1.6) ──────────────────────────────────

/// Recover an ECPoint from an ECDSA signature + digest + recovery id.
/// Returns null on any of the various failure modes (point at
/// infinity, x out of range, no square root, etc.); the caller
/// continues to the next recId. Algorithm parallels the bsvz
/// `recoverKeyFromSignature` Zig implementation.
ECPoint? _recoverPubkey(BigInt r, BigInt s, Uint8List digest, int recId) {
  final n = _secp256k1.n;
  final i = BigInt.from(recId ~/ 2);
  final x = r + (i * n);

  // x must be < curve prime p; if not, this recId is invalid.
  // pointycastle exposes the prime via the curve's q field (via cast).
  final curve = _secp256k1.curve;
  // For secp256k1 the order n and prime p are both close to 2^256;
  // out-of-range means r is beyond p. Try to decode the point — if it
  // fails, this recId is rejected.
  final yIsOdd = (recId & 1) == 1;
  final encoded = Uint8List(33);
  encoded[0] = yIsOdd ? 0x03 : 0x02;
  _writeBigIntBE(x, encoded, 1, 32);
  ECPoint? R;
  try {
    R = curve.decodePoint(encoded);
  } catch (_) {
    return null; // x not on curve / no square root
  }
  if (R == null) return null;

  // nR must equal infinity for a valid recovery; pointycastle's
  // ECPoint multiplication by n is not directly exposed; skip the
  // belt-and-braces check (the final bytes-equal comparison against
  // the expected pubkey discharges correctness).

  final e = _bytesToBigInt(digest);
  final rInv = r.modInverse(n);
  final Q =
      ((R * s)! + (_secp256k1.G * ((n - e) % n))!)! * rInv;
  return Q;
}

// ─── helpers ───────────────────────────────────────────────────────

BigInt _bytesToBigInt(List<int> bytes) {
  var n = BigInt.zero;
  for (final b in bytes) {
    n = (n << 8) | BigInt.from(b & 0xff);
  }
  return n;
}

void _writeBigIntBE(BigInt n, Uint8List out, int offset, int len) {
  var v = n;
  for (var i = len - 1; i >= 0; i--) {
    out[offset + i] = (v & BigInt.from(0xff)).toInt();
    v = v >> 8;
  }
}

bool _bytesEqual(List<int> a, List<int> b) {
  if (a.length != b.length) return false;
  for (var i = 0; i < a.length; i++) {
    if (a[i] != b[i]) return false;
  }
  return true;
}

/// D-O5m.followup-2 — Adapter-routed cell signer.
///
/// Wraps a `SecureSigningKeyAdapter` (Keychain-backed on iOS,
/// EncryptedSharedPreferences-backed on Android) + a `keyHandle`
/// so existing callers can opt into the secure path with a single
/// member call.  The adapter implementations forward to the same
/// underlying ECDSA-secp256k1-sha256 wire shape as
/// `signCellPayload(payloadBytes, devicePrivBytes)`, so the brain-
/// side verifier accepts a CellSigner-produced signature unchanged.
///
/// Backward compat: the `signCellPayload` top-level function stays
/// in place for legacy records that still hold the raw priv hex
/// in flutter_secure_storage.  The pairing service decides
/// per-record which path to take based on whether `secureKeyHandle`
/// is set in the persisted ChildCertRecord.
class CellSigner {
  final SecureSigningKeyAdapter _adapter;
  final String _keyHandle;

  /// Build a signer that routes through the supplied adapter for
  /// the priv stored at `keyHandle`.  The adapter + handle are
  /// captured at construction time so call sites don't have to
  /// re-pass them on every sign call.
  CellSigner({
    required SecureSigningKeyAdapter adapter,
    required String keyHandle,
  })  : _adapter = adapter,
        _keyHandle = keyHandle;

  /// Sign the canonical preimage bytes for a cell payload.  Returns
  /// 64 raw bytes (r || s, low-s normalised) — same shape as
  /// `signCellPayload(payloadBytes, devicePrivBytes)`.
  Future<Uint8List> signCanonicalCellPayload(Uint8List canonicalBytes) async {
    final sig =
        await _adapter.sign(keyHandle: _keyHandle, message: canonicalBytes);
    if (sig.length != cellSignatureLength) {
      throw StateError(
          'CellSigner.sign returned ${sig.length} bytes, expected $cellSignatureLength');
    }
    return sig;
  }

  /// Test-only accessor for the wrapped handle.  Used by the
  /// pairing-service tests to assert that a refactored call site
  /// is routing through the adapter instead of the raw-priv path.
  String get debugKeyHandle => _keyHandle;
}

/// Derive the 33-byte compressed-SEC1 pubkey for a 32-byte priv.
/// Convenience helper shared with the attachment_builder's
/// `capturedByCertId` derivation logic and the parity test.
Uint8List devicePubFromPriv(Uint8List devicePrivBytes) {
  if (devicePrivBytes.length != 32) {
    throw ArgumentError(
        'devicePrivBytes must be 32 bytes, got ${devicePrivBytes.length}');
  }
  final priv = ECPrivateKey(_bytesToBigInt(devicePrivBytes), _secp256k1);
  final pubPoint = (_secp256k1.G * priv.d!)!;
  return Uint8List.fromList(pubPoint.getEncoded(true));
}
