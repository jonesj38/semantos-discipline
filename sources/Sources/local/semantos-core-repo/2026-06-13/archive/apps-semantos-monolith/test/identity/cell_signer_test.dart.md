// D-O5m.followup-8 capture+upload — Cross-language ECDSA-secp256k1
// signing parity test.
//
// The fixture at test/fixtures/cell-signing-fixture.json is generated
// by `runtime/semantos-brain/tests/cell_signing_fixture_conformance.zig` (which
// runs as part of `zig build test`). This test asserts the Dart-side
// `signCellPayload` produces the SAME 64 bytes byte-for-byte for the
// same inputs — without that parity proof the brain might accept
// Dart-signed cells that brain-signed ones would reject (or vice
// versa).

import 'dart:convert';
import 'dart:io';
import 'dart:typed_data';

import 'package:test/test.dart';

import 'package:semantos/src/identity/cell_signer.dart';

Uint8List _hexToBytes(String hex) {
  final out = Uint8List(hex.length ~/ 2);
  for (var i = 0; i < out.length; i++) {
    out[i] = int.parse(hex.substring(i * 2, i * 2 + 2), radix: 16);
  }
  return out;
}

String _bytesToHex(List<int> bytes) {
  final sb = StringBuffer();
  for (final b in bytes) {
    sb.write((b & 0xff).toRadixString(16).padLeft(2, '0'));
  }
  return sb.toString();
}

void main() {
  group('cell_signer cross-language fixture parity', () {
    late Map<String, dynamic> fixture;

    setUpAll(() {
      // The fixture is generated by Zig and committed at this path.
      // Both `dart test` and the Semantos Brain test rig walk to the same file —
      // this is the load-bearing seam.
      final f = File('test/fixtures/cell-signing-fixture.json');
      if (!f.existsSync()) {
        throw StateError(
            'Fixture missing: ${f.absolute.path}. Run `zig build test` from runtime/semantos-brain/ to bootstrap.');
      }
      fixture = json.decode(f.readAsStringSync()) as Map<String, dynamic>;
    });

    test('signCellPayload produces byte-identical output to the Zig fixture', () {
      final priv = _hexToBytes(fixture['priv_hex'] as String);
      final payload = utf8.encode(fixture['payload_utf8'] as String);
      final expected = _hexToBytes(fixture['signature_hex'] as String);

      final sig = signCellPayload(payload, priv);

      expect(sig.length, equals(64));
      expect(_bytesToHex(sig), equals(_bytesToHex(expected)),
          reason:
              'Dart cell signer must produce byte-identical output to the Zig fixture; '
              'cross-language parity is load-bearing for the upload endpoint to accept '
              'Dart-signed cells.');
    });

    test('verifyCellSignature accepts the fixture signature', () {
      final pub = _hexToBytes(fixture['pub_hex'] as String);
      final payload = utf8.encode(fixture['payload_utf8'] as String);
      final sig = _hexToBytes(fixture['signature_hex'] as String);

      expect(verifyCellSignature(payload, sig, pub), isTrue);
    });

    test('verifyCellSignature rejects a tampered payload', () {
      final pub = _hexToBytes(fixture['pub_hex'] as String);
      final tampered = utf8.encode('not the original payload');
      final sig = _hexToBytes(fixture['signature_hex'] as String);

      expect(verifyCellSignature(tampered, sig, pub), isFalse);
    });

    test('verifyCellSignature rejects a tampered signature', () {
      final pub = _hexToBytes(fixture['pub_hex'] as String);
      final payload = utf8.encode(fixture['payload_utf8'] as String);
      final sig = _hexToBytes(fixture['signature_hex'] as String);
      // Flip a byte in r — recovery loop should fail to match the
      // expected pubkey.
      sig[0] ^= 0x01;

      expect(verifyCellSignature(payload, sig, pub), isFalse);
    });

    test('signCellPayload + verifyCellSignature round-trip on fresh inputs', () {
      // Generate signing material from a different priv to confirm
      // the signer isn't accidentally hard-coded to the fixture.
      final priv = _hexToBytes(
          'a1b2c3d4e5f600112233445566778899aabbccddeeff00112233445566778899');
      final pub = devicePubFromPriv(priv);
      final payload = utf8.encode('arbitrary cell payload bytes');

      final sig = signCellPayload(payload, priv);
      expect(verifyCellSignature(payload, sig, pub), isTrue);
      expect(sig.length, equals(64));
    });

    test('signCellPayload is deterministic — same inputs → same bytes', () {
      final priv = _hexToBytes(fixture['priv_hex'] as String);
      final payload = utf8.encode('deterministic test');
      final a = signCellPayload(payload, priv);
      final b = signCellPayload(payload, priv);
      expect(_bytesToHex(a), equals(_bytesToHex(b)));
    });

    test('signCellPayload rejects non-32-byte priv', () {
      expect(
        () => signCellPayload(utf8.encode('x'), Uint8List(31)),
        throwsArgumentError,
      );
      expect(
        () => signCellPayload(utf8.encode('x'), Uint8List(33)),
        throwsArgumentError,
      );
    });
  });
}
