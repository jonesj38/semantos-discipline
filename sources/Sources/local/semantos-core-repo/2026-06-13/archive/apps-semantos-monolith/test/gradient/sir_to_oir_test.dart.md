// D-O5m.followup-3 Phase 3 — sir_to_oir unit + parity tests.
//
// Reference: apps/oddjobz-mobile/lib/src/gradient/sir_to_oir.dart
//            (the unit under test, a verbatim port of
//            core/semantos-sir/src/lower-sir.ts);
//            apps/oddjobz-mobile/test/fixtures/sir-to-oir-fixture.json
//            (the cross-language fixture generated by
//            runtime/intent/scripts/gen-phase3-fixtures.ts -- the
//            authoritative TS-side OIR output the Dart port must
//            match byte-for-byte).

import 'dart:convert' show jsonDecode;
import 'dart:io';

import 'package:semantos/src/gradient/sir_to_oir.dart';
import 'package:test/test.dart';

Map<String, dynamic> _gov({
  String trustClass = 'interpretive',
  String proofRequirement = 'attestation',
  String executionAuthority = 'hat_scoped',
  String linearity = 'LINEAR',
  List<String>? allowedEmitOps,
  Map<String, dynamic>? domainBinding,
}) =>
    {
      'trustClass': trustClass,
      'proofRequirement': proofRequirement,
      'executionAuthority': executionAuthority,
      'linearity': linearity,
      if (allowedEmitOps != null) 'allowedEmitOps': allowedEmitOps,
      if (domainBinding != null) 'domainBinding': domainBinding,
    };

Map<String, dynamic> _node(String category, Map<String, dynamic> rest) {
  return {
    'id': '\$s0',
    // Tagged jural category triggers the per-category lowering path
    // in sirToOir; the untagged-string convention falls through to
    // constraint-only (covered separately in the fixture parity test).
    'category': {'lexicon': 'jural', 'category': category},
    'taxonomy': const {'what': 'demo', 'how': 'demo', 'why': 'demo'},
    'identity': const {
      'subject': {'type': 'role', 'name': 'demo'},
    },
    'governance': _gov(),
    'action': 'demo',
    'provenance': const {
      'source': 'voice',
      'expressedAt': '2026-04-17T00:00:00Z',
      'trustAtExpression': 'interpretive',
    },
    ...rest,
  };
}

Map<String, dynamic> _program(Map<String, dynamic> node,
    {Map<String, dynamic>? programGov}) {
  return {
    'nodes': [node],
    'primaryNodeId': node['id'],
    'programGovernance': programGov ?? _gov(),
  };
}

void main() {
  group('sirToOir — happy paths (one per jural category)', () {
    test('declaration -> constraint-only', () {
      final program = _program(_node('declaration', {
        'constraint': {
          'kind': 'value',
          'field': 'status',
          'op': '=',
          'value': 'executed',
        },
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      expect(p.bindings, hasLength(1));
      expect(p.bindings.first.kind, equals('comparison'));
    });

    test('obligation with deadline -> constraint + timeBefore + and', () {
      final program = _program(_node('obligation', {
        'constraint': {'kind': 'capability', 'required': 5, 'name': 'METERING'},
        'fulfillment': {'fulfilledBy': 'evt', 'deadline': '2026-04-22T17:00:00Z'},
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      final kinds = p.bindings.map((b) => b.kind).toList();
      expect(kinds, contains('capability'));
      expect(kinds, contains('timeConstraint'));
      expect(kinds, contains('logical_and'));
    });

    test('permission -> single capability binding', () {
      final program = _program(_node('permission', {
        'constraint': {'kind': 'capability', 'required': 3, 'name': 'OPERATE'},
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      expect(p.bindings, hasLength(1));
      expect(p.bindings.first.kind, equals('capability'));
      expect(p.bindings.first.capabilityNumber, equals(3));
    });

    test('prohibition -> constraint + logical_not', () {
      final program = _program(_node('prohibition', {
        'constraint': {
          'kind': 'value',
          'field': 'pressure',
          'op': '>',
          'value': 150,
        },
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      expect(p.bindings, hasLength(2));
      expect(p.bindings[1].kind, equals('logical_not'));
    });

    test('power with typeHash -> capability + typeHashCheck + and', () {
      final program = _program(_node('power', {
        'constraint': {'kind': 'capability', 'required': 11, 'name': 'PUBLISH'},
        'target': {
          'typeHash':
              '1a3771053c73eca4ec4c5ef0c662811117e58a5ed1e49d499fda5ac37b7a0afd',
        },
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      final kinds = p.bindings.map((b) => b.kind).toSet();
      expect(kinds, contains('capability'));
      expect(kinds, contains('typeHashCheck'));
      expect(kinds, contains('logical_and'));
    });

    test('condition with temporal gate -> single timeAfter', () {
      final program = _program(_node('condition', {
        'constraint': {'kind': 'temporal', 'op': 'after', 'iso': '2026-04-22T00:00:00Z'},
        'gate': {'type': 'temporal', 'deadline': '2026-04-22T00:00:00Z'},
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      expect(p.bindings, hasLength(1));
      expect(p.bindings.first.kind, equals('timeConstraint'));
      expect(p.bindings.first.timeOp, equals('timeAfter'));
    });

    test('transfer -> constraint passthrough', () {
      final program = _program(_node('transfer', {
        'constraint': {
          'kind': 'composite',
          'op': 'and',
          'children': [
            {'kind': 'capability', 'required': 9, 'name': 'TRANSFER'},
            {'kind': 'capability', 'required': 5, 'name': 'METERING'},
          ],
        },
        'transferTo': {
          'subject': {'type': 'domainFlag', 'flag': 4},
        },
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirSuccess>());
      final p = (r as SirToOirSuccess).program;
      final caps = p.bindings.where((b) => b.kind == 'capability').toList();
      expect(caps, hasLength(2));
    });
  });

  group('sirToOir — refusals', () {
    test('authoritative + non-formal -> trustTierMismatch', () {
      final program = _program(
        _node('power', {
          'governance': _gov(
            trustClass: 'authoritative',
            proofRequirement: 'attestation',
          ),
          'constraint': {'kind': 'capability', 'required': 5, 'name': 'PUB'},
        }),
        programGov: _gov(
          trustClass: 'authoritative',
          proofRequirement: 'attestation',
        ),
      );
      final r = sirToOir(program);
      expect(r, isA<SirToOirRejected>());
      expect((r as SirToOirRejected).rejection.error,
          equals(LoweringError.trustTierMismatch));
      expect(r.rejection.code, equals('TRUST_TIER_VIOLATION'));
    });

    test('delegated authority -> delegationUnconfigured', () {
      final program = _program(
        _node('power', {
          'governance': _gov(executionAuthority: 'delegated'),
          'constraint': {'kind': 'capability', 'required': 5, 'name': 'PUB'},
        }),
        programGov: _gov(executionAuthority: 'delegated'),
      );
      final r = sirToOir(program);
      expect(r, isA<SirToOirRejected>());
      expect((r as SirToOirRejected).rejection.error,
          equals(LoweringError.delegationUnconfigured));
    });

    test('allowedEmitOps mismatch -> emitOpNotAllowed', () {
      final program = _program(_node('prohibition', {
        'governance': _gov(allowedEmitOps: ['comparison']),
        'constraint': {
          'kind': 'value',
          'field': 'pressure',
          'op': '>',
          'value': 150,
        },
      }));
      final r = sirToOir(program);
      expect(r, isA<SirToOirRejected>());
      final rej = (r as SirToOirRejected).rejection;
      expect(rej.error, equals(LoweringError.emitOpNotAllowed));
      expect(rej.message, contains('logical_not'));
    });

    test('missing primary node -> primaryNodeNotFound', () {
      final program = {
        'nodes': const [],
        'primaryNodeId': '\$s99',
        'programGovernance': _gov(),
      };
      final r = sirToOir(program);
      expect(r, isA<SirToOirRejected>());
      expect((r as SirToOirRejected).rejection.error,
          equals(LoweringError.primaryNodeNotFound));
    });
  });

  group('sirToOir — cross-language fixture parity', () {
    test('every fixture case produces byte-identical OIR JSON', () {
      final fixturePath = _findFixture('sir-to-oir-fixture.json');
      final fixture = jsonDecode(File(fixturePath).readAsStringSync())
          as Map<String, dynamic>;
      final cases = fixture['cases'] as List;
      expect(cases, isNotEmpty);
      for (final raw in cases) {
        final c = raw as Map<String, dynamic>;
        final name = c['name'] as String;
        final sirProgram = c['sirProgram'] as Map<String, dynamic>;
        final expectedJson = c['canonicalOirJson'] as String;
        final r = sirToOir(sirProgram);
        expect(r, isA<SirToOirSuccess>(), reason: 'case $name lowered');
        final actualJson =
            canonicalizeJson((r as SirToOirSuccess).program.toJson());
        expect(actualJson, equals(expectedJson),
            reason: 'OIR JSON parity for case $name');
      }
    });
  });
}

/// Walk up from the test's CWD looking for the fixture file. Allows
/// the same test to run from `dart test` (cwd = oddjobz-mobile) or
/// from the repo root.
String _findFixture(String filename) {
  var dir = Directory.current;
  for (var i = 0; i < 8; i++) {
    final p =
        '${dir.path}/apps/oddjobz-mobile/test/fixtures/$filename';
    if (File(p).existsSync()) return p;
    final localP = '${dir.path}/test/fixtures/$filename';
    if (File(localP).existsSync()) return localP;
    final parent = dir.parent;
    if (parent.path == dir.path) break;
    dir = parent;
  }
  throw StateError('fixture $filename not found from ${Directory.current.path}');
}
