/// A single OIR binding -- mirrors `IRBinding` in
/// core/semantos-ir/src/types.ts. Stored as a typed class so the
/// emitter can match on `kind` exhaustively.
class OirBinding {
  final String name;
  final String kind;
  // Comparison fields:
  final String? op;
  final String? field;
  // Comparison value: int | String
  final Object? value;
  // Logical/composite:
  final List<String>? operands;
  // Capability:
  final int? capabilityNumber;
  // Domain:
  final Object? domainFlag; // int | String
  // Time:
  final String? timeOp;
  final int? timestamp;
  // Host call:
  final String? functionName;
  // Type-hash:
  final String? expectedHash;

  const OirBinding({
    required this.name,
    required this.kind,
    this.op,
    this.field,
    this.value,
    this.operands,
    this.capabilityNumber,
    this.domainFlag,
    this.timeOp,
    this.timestamp,
    this.functionName,
    this.expectedHash,
  });

  /// Decode an [OirBinding] from its JSON shape (mirrors
  /// `IRBinding` in core/semantos-ir/src/types.ts). Used by the
  /// cross-language fixture parity tests.
  static OirBinding fromJson(Map<String, dynamic> j) {
    final operands = j['operands'];
    return OirBinding(
      name: j['name'] as String,
      kind: j['kind'] as String,
      op: j['op'] as String?,
      field: j['field'] as String?,
      value: j['value'],
      operands: operands is List ? operands.cast<String>() : null,
      capabilityNumber: (j['capabilityNumber'] as num?)?.toInt(),
      domainFlag: j['domainFlag'],
      timeOp: j['timeOp'] as String?,
      timestamp: (j['timestamp'] as num?)?.toInt(),
      functionName: j['functionName'] as String?,
      expectedHash: j['expectedHash'] as String?,
    );
  }

  /// Encode to a `Map<String, dynamic>` matching the TS canonical
  /// JSON shape -- only set fields are emitted, no `null` filler.
  Map<String, dynamic> toJson() {
    final out = <String, dynamic>{'name': name, 'kind': kind};
    if (op != null) out['op'] = op;
    if (field != null) out['field'] = field;
    if (value != null) out['value'] = value;
    if (operands != null) out['operands'] = operands;
    if (capabilityNumber != null) out['capabilityNumber'] = capabilityNumber;
    if (domainFlag != null) out['domainFlag'] = domainFlag;
    if (timeOp != null) out['timeOp'] = timeOp;
    if (timestamp != null) out['timestamp'] = timestamp;
    if (functionName != null) out['functionName'] = functionName;
    if (expectedHash != null) out['expectedHash'] = expectedHash;
    return out;
  }
}

/// A complete OIR program -- mirrors `IRProgram` in
/// core/semantos-ir/src/types.ts.
class OirProgram {
  final List<OirBinding> bindings;
  final String result;

  const OirProgram({required this.bindings, required this.result});

  static OirProgram fromJson(Map<String, dynamic> j) {
    final raw = j['bindings'] as List;
    return OirProgram(
      bindings: raw
          .map((e) => OirBinding.fromJson(e as Map<String, dynamic>))
          .toList(),
      result: j['result'] as String,
    );
  }

  Map<String, dynamic> toJson() => {
        'bindings': bindings.map((b) => b.toJson()).toList(),
        'result': result,
      };
}

/// Structured rejection from `sirToOir`. Mirrors the TS
/// `LoweringResult` rejection shape (`{ok:false, code, message}`) plus
/// a typed enum so the Dart pipeline can route by stage.
enum LoweringError {
  trustTierMismatch,
  delegationUnconfigured,
  emitOpNotAllowed,
  primaryNodeNotFound,
  identityCapMismatch,
  invalidConstraint,
  invalidNode,
}

class LoweringRejection {
  final LoweringError error;
  final String code;
  final String message;
  const LoweringRejection({
    required this.error,
    required this.code,
    required this.message,
  });
}

  const SirToOirSuccess(this.program);

class SirToOirRejected extends SirToOirResult {

class _NameGen {

// ── Trust-tier enforcement ──────────────────────────────────────

LoweringRejection? _enforceTrustTier(Map<String, dynamic> governance) {
  final trustClass = governance['trustClass'];
  final proofRequirement = governance['proofRequirement'];
  final executionAuthority = governance['executionAuthority'];
  if (trustClass == 'authoritative' && proofRequirement != 'formal') {
    return const LoweringRejection(
      error: LoweringError.trustTierMismatch,
      code: 'TRUST_TIER_VIOLATION',
      message:
          'Cannot lower authoritative expression without formal proof requirement',
    );
  }
  if (executionAuthority == 'delegated') {
    return const LoweringRejection(
      error: LoweringError.delegationUnconfigured,
      code: 'DELEGATED_NOT_IMPLEMENTED',
      message: 'Delegated execution authority not yet implemented',
    );
  }
  return null;
}

// ── Helpers ─────────────────────────────────────────────────────

/// Returns `true` when [name] points to a `logical_and` / `logical_or`
/// binding with fewer than 2 operands — i.e. one that the byte emitter
/// translates into zero opcodes. Used by the domain-binding wrap so a
/// constraint-free SIR node doesn't generate a dangling OP_BOOLAND.
bool _isEmptyLogicalCombinator(List<OirBinding> bindings, String name) {
  for (final b in bindings) {
    if (b.name != name) continue;
    final isCombinator =
        b.kind == 'logical_and' || b.kind == 'logical_or';
    if (!isCombinator) return false;
    final n = b.operands?.length ?? 0;
    return n < 2;
  }
  return false;
}

// ── AllowedEmitOps enforcement ─────────────────────────────────

LoweringRejection? _enforceAllowedEmitOps(
  List<OirBinding> bindings,
  List<String> allowedEmitOps,
  String nodeId,
) {
  for (final b in bindings) {
    if (!allowedEmitOps.contains(b.kind)) {
      return LoweringRejection(
        error: LoweringError.emitOpNotAllowed,
        code: 'EMIT_OP_NOT_ALLOWED',
        message:
            "OIR binding kind '${b.kind}' not in allowedEmitOps for $nodeId. "
            'Whitelist: [${allowedEmitOps.join(', ')}]',
      );
    }
  }
  return null;
}

/// Lower a SIR constraint to OIR bindings. Returns the binding name
/// of the result (so parent expressions can reference it).
String _lowerConstraint(
  Map<String, dynamic> constraint,
  _NameGen names,
  List<OirBinding> bindings,
) {
  // Defensive: if the SIR producer (llama / Claude / hand-built fixture)
  // emitted a constraint without a `kind` field — or with an unknown
  // kind — fall back to a no-op `capability(0)` binding so the
  // pipeline reaches the kernel with a structurally-valid OIR. The
  // kernel will reject capability(0) as held-but-not-meaningful, so
  // we don't silently grant authority; we just give the trace a
  // chance to render so the user can see WHY their intent didn't
  // execute (instead of crashing the whole pipeline with a null-cast).
  final rawKind = constraint['kind'];
  if (rawKind is! String) {
    debugPrint(
      '[sir_to_oir] constraint missing "kind" field — substituting noop. '
      'Got: $constraint',
    );
    final name = names.next();
    bindings.add(OirBinding(
      name: name,
      kind: 'capability',
      capabilityNumber: 0,
    ));
    return name;
  }
  final kind = rawKind;
  switch (kind) {
    case 'capability':
      {
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'capability',
          capabilityNumber: (constraint['required'] as num).toInt(),
        ));
        return name;
      }
    case 'domain':
      {
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'domainCheck',
          domainFlag: constraint['flag'],
        ));
        return name;
      }
    case 'identity':
      {
        final ref = constraint['ref'] as Map<String, dynamic>?;
        final type = ref?['type'];
        if (type == 'domainFlag') {
          final name = names.next();
          bindings.add(OirBinding(
            name: name,
            kind: 'domainCheck',
            domainFlag: ref!['flag'],
          ));
          return name;
        }
        // Role / certPattern -> placeholder domainCheck (governance
        // plane resolves the actual identity binding).
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'domainCheck',
          domainFlag: type == 'role' ? (ref!['name'] as String) : 0,
        ));
        return name;
      }
    case 'temporal':
      {
        final name = names.next();
        final iso = constraint['iso'] as String;
        final timestamp = DateTime.parse(iso).millisecondsSinceEpoch ~/ 1000;
        bindings.add(OirBinding(
          name: name,
          kind: 'timeConstraint',
          timeOp: constraint['op'] == 'after' ? 'timeAfter' : 'timeBefore',
          timestamp: timestamp,
        ));
        return name;
      }
    case 'value':
      {
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'comparison',
          op: constraint['op'] as String,
          field: constraint['field'] as String,
          value: constraint['value'],
        ));
        return name;
      }
    case 'state':
      {
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'comparison',
          op: '=',
          field: 'phase',
          value: constraint['requiredPhase'],
        ));
        return name;
      }
    case 'interlock':
      {
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: 'hostCall',
          functionName: 'interlock:${constraint['policyId']}',
        ));
        return name;
      }
    case 'composite':
      {
        final children = constraint['children'] as List;
        final childNames = <String>[];
        for (final c in children) {
          childNames.add(_lowerConstraint(
              c as Map<String, dynamic>, names, bindings));
        }
        if (constraint['op'] == 'not') {
          final name = names.next();
          bindings.add(OirBinding(
            name: name,
            kind: 'logical_not',
            operands: [childNames.first],
          ));
          return name;
        }
        final logicalKind =
            constraint['op'] == 'and' ? 'logical_and' : 'logical_or';
        final name = names.next();
        bindings.add(OirBinding(
          name: name,
          kind: logicalKind,
          operands: childNames,
        ));
        return name;
      }
  }
  throw StateError('unreachable: unhandled constraint kind $kind');
}

/// Lower a SIR node to OIR bindings based on its (lexicon, category)
/// pair. Mirrors core/semantos-sir/src/lower-sir.ts:lowerCategory.
String _lowerCategory(
  Map<String, dynamic> node,
  _NameGen names,
  List<OirBinding> bindings,
) {
  final category = node['category'];
  // Mirror lower-sir.ts: only `node.category.lexicon === 'jural'`
  // takes the per-category bespoke lowering path. Everything else
  // (untagged-string categories, non-jural tagged categories) falls
  // back to constraint-only lowering -- the safe default that
  // preserves the constraint semantics without inventing rules.
  String? lexicon;
  String? cat;
  if (category is Map) {
    lexicon = category['lexicon'] as String?;
    cat = category['category'] as String?;
  }
  final constraint = node['constraint'] as Map<String, dynamic>;
  if (lexicon != 'jural') {
    return _lowerConstraint(constraint, names, bindings);
  }
  switch (cat) {
    case 'declaration':
      // Declaration -> identity/domain check + constraint assertions.
      return _lowerConstraint(constraint, names, bindings);
    case 'obligation':
      {
        final parts = <String>[
          _lowerConstraint(constraint, names, bindings),
        ];
        final fulfillment = node['fulfillment'] as Map<String, dynamic>?;
        final deadline = fulfillment?['deadline'] as String?;
        if (deadline != null) {
          final deadlineName = names.next();
          final ts = DateTime.parse(deadline).millisecondsSinceEpoch ~/ 1000;
          bindings.add(OirBinding(
            name: deadlineName,
            kind: 'timeConstraint',
            timeOp: 'timeBefore',
            timestamp: ts,
          ));
          parts.add(deadlineName);
        }
        if (parts.length == 1) return parts.first;
        final andName = names.next();
        bindings.add(OirBinding(
          name: andName,
          kind: 'logical_and',
          operands: parts,
        ));
        return andName;
      }
    case 'permission':
      return _lowerConstraint(constraint, names, bindings);
    case 'prohibition':
      {
        final innerName = _lowerConstraint(constraint, names, bindings);
        final notName = names.next();
        bindings.add(OirBinding(
          name: notName,
          kind: 'logical_not',
          operands: [innerName],
        ));
        return notName;
      }
    case 'power':
      {
        final parts = <String>[
          _lowerConstraint(constraint, names, bindings),
        ];
        final target = node['target'] as Map<String, dynamic>?;
        final typeHash = target?['typeHash'] as String?;
        if (typeHash != null) {
          final hashName = names.next();
          bindings.add(OirBinding(
            name: hashName,
            kind: 'typeHashCheck',
            expectedHash: typeHash,
          ));
          parts.add(hashName);
        }
        if (parts.length == 1) return parts.first;
        final andName = names.next();
        bindings.add(OirBinding(
          name: andName,
          kind: 'logical_and',
          operands: parts,
        ));
        return andName;
      }
    case 'condition':
      {
        final gate = node['gate'] as Map<String, dynamic>?;
        if (gate != null) {
          final gateType = gate['type'];
          if (gateType == 'temporal' && gate['deadline'] is String) {
            final name = names.next();
            final ts = DateTime.parse(gate['deadline'] as String)
                    .millisecondsSinceEpoch ~/
                1000;
            bindings.add(OirBinding(
              name: name,
              kind: 'timeConstraint',
              timeOp: 'timeAfter',
              timestamp: ts,
            ));
            return name;
          }
          if (gateType == 'state' && gate['requiredPhase'] is String) {
            final name = names.next();
            bindings.add(OirBinding(
              name: name,
              kind: 'comparison',
              op: '=',
              field: 'phase',
              value: gate['requiredPhase'],
            ));
            return name;
          }
          if (gateType == 'value' && gate['threshold'] is Map) {
            final name = names.next();
            final t = gate['threshold'] as Map<String, dynamic>;
            bindings.add(OirBinding(
              name: name,
              kind: 'comparison',
              op: t['op'] as String,
              field: t['field'] as String,
              value: t['value'],
            ));
            return name;
          }
        }
        return _lowerConstraint(constraint, names, bindings);
      }
    case 'transfer':
      {
        final parts = <String>[
          _lowerConstraint(constraint, names, bindings),
        ];
        if (parts.length == 1) return parts.first;
        final andName = names.next();
        bindings.add(OirBinding(
          name: andName,
          kind: 'logical_and',
          operands: parts,
        ));
        return andName;
      }
    default:
      // Unknown category -> safe default (constraint-only).
      return _lowerConstraint(constraint, names, bindings);
  }
}

/// Lower a SIR program (as a `Map<String, dynamic>` matching the TS
/// SIRProgram shape) to an OIR program. Returns a structured result
/// -- never throws.
SirToOirResult sirToOir(Map<String, dynamic> program) {
  // Find primary node.
  final nodes = program['nodes'] as List?;
  final primaryId = program['primaryNodeId'];
  if (nodes == null || primaryId == null) {
    return const SirToOirRejected(LoweringRejection(
      error: LoweringError.invalidNode,
      code: 'INVALID_PROGRAM',
      message: 'program missing nodes or primaryNodeId',
    ));
  }
  Map<String, dynamic>? primary;
  for (final n in nodes) {
    if (n is Map && n['id'] == primaryId) {
      primary = Map<String, dynamic>.from(n);
      break;
    }
  }
  if (primary == null) {
    return SirToOirRejected(LoweringRejection(
      error: LoweringError.primaryNodeNotFound,
      code: 'PRIMARY_NODE_NOT_FOUND',
      message: "No node with id '$primaryId' in program",
    ));
  }

  final programGov = program['programGovernance'];
  if (programGov is Map<String, dynamic>) {
    final err = _enforceTrustTier(programGov);
    if (err != null) return SirToOirRejected(err);
  }

  final nodeGov = primary['governance'];
  if (nodeGov is Map<String, dynamic>) {
    final err = _enforceTrustTier(nodeGov);
    if (err != null) return SirToOirRejected(err);
  }

  // Lower the primary node.
  final names = _NameGen();
  final bindings = <OirBinding>[];
  String result = _lowerCategory(primary, names, bindings);

  // Domain binding enforcement: emit domainCheck for bound flag (and
  // parent). When the constraint side is empty (SIR producer emitted
  // constraints=[]), we cannot legitimately run OP_CHECKDOMAINFLAG
  // either, because the kernel requires a cell to be on the stack
  // first and no script-level opcode exists to push the "self" cell.
  // (This is a known substrate gap — every runtime/intent test
  // currently mocks the kernel verdict, so the script→kernel cell
  // bridge has never been exercised end-to-end.)
  //
  // For now, when the constraint side is empty we emit a single
  // `literal_true` binding (OP_1 / OP_TRUE) — the producer is the
  // sole gatekeeper, the kernel script is vacuous. This lets the
  // pipeline mint the cell so the inspector cascade completes;
  // when the substrate gains a cell-push opcode we can resurrect
  // the full domainCheck path.
  final db = nodeGov is Map<String, dynamic>
      ? nodeGov['domainBinding'] as Map<String, dynamic>?
      : null;
  final constraintIsEmpty = _isEmptyLogicalCombinator(bindings, result);
  if (constraintIsEmpty) {
    bindings.removeWhere((b) => b.name == result);
    final tName = names.next();
    bindings.add(OirBinding(name: tName, kind: 'literal_true'));
    result = tName;
  } else if (db != null && db['flag'] != null) {
    final childName = names.next();
    bindings.add(OirBinding(
      name: childName,
      kind: 'domainCheck',
      domainFlag: db['flag'],
    ));
    String? parentName;
    if (db['parentFlag'] != null) {
      parentName = names.next();
      bindings.add(OirBinding(
        name: parentName,
        kind: 'domainCheck',
        domainFlag: db['parentFlag'],
      ));
    }
    final domainParts = <String>[result, childName];
    if (parentName != null) domainParts.add(parentName);
    final andName = names.next();
    bindings.add(OirBinding(
      name: andName,
      kind: 'logical_and',
      operands: domainParts,
    ));
    result = andName;
  }

  // AllowedEmitOps enforcement.
  if (nodeGov is Map<String, dynamic>) {
    final allowed = nodeGov['allowedEmitOps'];
    if (allowed is List) {
      final err = _enforceAllowedEmitOps(
        bindings,
        allowed.cast<String>(),
        primary['id'] as String,
      );
      if (err != null) return SirToOirRejected(err);
    }
  }

  return SirToOirSuccess(OirProgram(bindings: bindings, result: result));
}

/// Build a SIR program from a Phase-2 Intent map + a HatContext.
/// Mirrors `runtime/intent/src/sir-builder.ts:buildSIR` in the simple
/// path the on-device pipeline takes (no formal-proof tiering, no
/// authority binding, no delegation -- those flow through the brain
/// path when they apply).
///
/// Pure function -- given the same Intent + hat fields, produces the
/// same SIRProgram.
Map<String, dynamic> buildSir({
  required Map<String, dynamic> intent,
  required String hatId,
  String? certId,
  required int domainFlag,
  required String maxTrustClass,
  String? extensionId,
  DateTime? expressedAt,
}) {
  final canonical = canonicaliseIntent(intent);
  // Trust-class clamping mirrors candidateTrustClass + minTrustClass
  // in sir-builder.ts.
  String trustClass;
  final source = canonical['source'];
  if (source == 'shell' || source == 'host-exec' || source == 'network' ||
      source == 'scheduler') {
    trustClass = 'interpretive';
  } else {
    final c = (canonical['confidence'] as num?)?.toDouble() ?? 0.0;
    if (c >= 0.9) {
      trustClass = 'interpretive';
    } else {
      trustClass = 'cosmetic';
    }
  }
  // Cap by hat ceiling.
  final order = const ['cosmetic', 'interpretive', 'authoritative'];
  final candidateRank = order.indexOf(trustClass);
  final ceilingRank = order.indexOf(maxTrustClass);
  if (candidateRank > ceilingRank) {
    trustClass = maxTrustClass;
  }

  final governance = <String, dynamic>{
    'trustClass': trustClass,
    'proofRequirement': trustClass == 'authoritative' ? 'formal' : 'none',
    'executionAuthority': 'hat_scoped',
    'linearity': 'AFFINE',
    'domainBinding': {
      'flag': domainFlag,
      'domainType': 'personal',
    },
  };
  final identity = <String, dynamic>{
    'subject': {'type': 'role', 'name': hatId},
    'hatId': hatId,
    if (certId != null) 'certId': certId,
  };
  final provenance = <String, dynamic>{
    'source': _sourceToProvenance(source as String? ?? 'voice'),
    'confidence': canonical['confidence'] ?? 0.0,
    'expressedAt': (expressedAt ?? DateTime.now().toUtc())
        .toIso8601String(),
    'trustAtExpression': trustClass,
  };

  final constraints = (canonical['constraints'] as List?) ?? const [];
  // Empty constraints lowers to an empty `logical_and` OIR binding
  // (operands=[]) which emits zero bytes. The downstream domain-
  // binding wrap detects this case via _isEmptyComposite and skips
  // the outer logical_and (which would otherwise emit a dangling
  // OP_BOOLAND that the kernel rejects).
  final consolidated = constraints.isEmpty
      ? {'kind': 'composite', 'op': 'and', 'children': []}
      : constraints.length == 1
          ? Map<String, dynamic>.from(
              constraints.first as Map<String, dynamic>)
          : {
              'kind': 'composite',
              'op': 'and',
              'children':
                  constraints.map((c) => c as Map<String, dynamic>).toList(),
            };

  final node = <String, dynamic>{
    'id': '\$s0',
    'category': canonical['category'],
    'taxonomy': canonical['taxonomy'],
    'identity': identity,
    'governance': governance,
    'action': canonical['action'],
    'constraint': consolidated,
    if (canonical['target'] != null) 'target': canonical['target'],
    if (canonical['transferTo'] != null) 'transferTo': canonical['transferTo'],
    if (canonical['fulfillment'] != null)
      'fulfillment': canonical['fulfillment'],
    'provenance': provenance,
  };

  return {
    'nodes': [node],
    'primaryNodeId': '\$s0',
    'programGovernance': governance,
  };
}

/// RFC 8785 canonical JSON of an arbitrary value -- mirrors
/// `core/semantos-ir/src/canonical.ts:canonicalize`. Used by the
/// cross-language α-equivalence tests to assert the Dart pipeline
/// produces identical OIR JSON to the TS pipeline for any well-formed
/// SIR program.
String canonicalizeJson(Object? value) {
  if (value == null) return 'null';
  if (value is bool) return value ? 'true' : 'false';
  if (value is num) {
    if (!value.isFinite) return 'null';
    if (value is int) return value.toString();
    // Match ES2015 Number.prototype.toString semantics for ints.
    if (value == value.truncateToDouble()) return value.toInt().toString();
    return value.toString();
  }
  if (value is String) return _jsonEncodeString(value);
  if (value is List) {
    return '[${value.map(canonicalizeJson).join(',')}]';
  }
  if (value is Map) {
    final keys = value.keys
        .where((k) => value[k] != null)
        .map((k) => k.toString())
        .toList()
      ..sort();
    final pairs = keys
        .map((k) => '${_jsonEncodeString(k)}:${canonicalizeJson(value[k])}');
    return '{${pairs.join(',')}}';
  }
  return 'null';
}

String _jsonEncodeString(String s) {
  final sb = StringBuffer('"');
  for (final r in s.runes) {
    switch (r) {
      case 0x22:
        sb.write('\\"');
        break;
      case 0x5C:
        sb.write('\\\\');
        break;
      case 0x08:
        sb.write('\\b');
        break;
      case 0x09:
        sb.write('\\t');
        break;
      case 0x0A:
        sb.write('\\n');
        break;
      case 0x0C:
        sb.write('\\f');
        break;
      case 0x0D:
        sb.write('\\r');
        break;
      default:
        if (r < 0x20) {
          sb.write('\\u${r.toRadixString(16).padLeft(4, '0')}');
        } else {
          sb.writeCharCode(r);
        }
    }
  }
  sb.write('"');
  return sb.toString();
}

String _sourceToProvenance(String source) {
  switch (source) {
    case 'nl':
    case 'ui':
      return 'manual';
    case 'voice':
      return 'voice';
    case 'shell':
    case 'host-exec':
      return 'api';
    case 'network':
    case 'governance':
      return 'monitor';
    case 'scheduler':
      return 'scheduler';
    default:
      return 'manual';
  }
}
