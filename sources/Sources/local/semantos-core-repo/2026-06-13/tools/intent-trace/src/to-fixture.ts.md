/**
 * RM-094 — JSONL trace → TS regression-test fixture.
 *
 * Captures the *structural fingerprint* of a single correlation group
 * (stage names, pass order, contributionKeys, skipInComposite markers,
 * alternativesCount) and emits a Bun test file that:
 *
 *   1. Imports a named reducer fixture (`T1_REPORT_DRIPPING_TAP`, etc.).
 *   2. Re-runs `reduceToIntent` against that fixture with an in-memory
 *      logger.
 *   3. Asserts the new event sequence has the same structural
 *      fingerprint as the captured trace.
 *
 * Volatile fields (ts, durationMs, correlationId, intentId) are not
 * asserted — the fingerprint is the structural shape, not the timing.
 *
 * No LLM, no fixtures sniffing — the caller names the input fixture
 * explicitly (e.g. `intent-trace fixturize trace.jsonl --input
 * T1_REPORT_DRIPPING_TAP > regression.test.ts`). This keeps the
 * tool deterministic and the produced test self-describing.
 */

import { groupByCorrelation, parseTrace, type TraceEvent } from './parse.js';

export interface FixtureEvent {
  stage: string;
  /** Subset of `data` fields we treat as structurally significant. */
  data: {
    pass?: string;
    skipInComposite?: boolean;
    alternativesCount?: number;
    contributionKeys?: ReadonlyArray<string>;
    reason?: unknown; // for intent_rejected
  };
}

export interface ToFixtureOptions {
  /** Name of an exported reducer fixture (e.g. `T1_REPORT_DRIPPING_TAP`).
   *  The generated test imports `<fixtureName>` from
   *  `@semantos/intent/reducer/__fixtures__/trades-fixtures` by default. */
  inputFixtureName: string;
  /** Override the import path. Default: relative to a typical test in
   *  `runtime/intent/src/__tests__/`. Override when emitting the test
   *  into a different package. */
  fixturesImport?: string;
  /** Optional `correlationId` to filter the trace by. Defaults to the
   *  first correlation group found. */
  correlationId?: string;
}

/** Extract the structural fingerprint from a trace event. */
export function fingerprintEvent(e: TraceEvent): FixtureEvent {
  const d = e.data as Record<string, unknown>;
  const fp: FixtureEvent['data'] = {};
  if (typeof d.pass === 'string') fp.pass = d.pass;
  if (typeof d.skipInComposite === 'boolean') fp.skipInComposite = d.skipInComposite;
  if (typeof d.alternativesCount === 'number') fp.alternativesCount = d.alternativesCount;
  if (Array.isArray(d.contributionKeys)) {
    fp.contributionKeys = (d.contributionKeys as unknown[]).map(String);
  }
  if (d.reason !== undefined) fp.reason = d.reason;
  return { stage: e.stage, data: fp };
}

/** Build the structural fingerprint of a correlation group. */
export function fingerprintTrace(events: TraceEvent[]): FixtureEvent[] {
  return events.map(fingerprintEvent);
}

/** Emit the TS source of a regression test that asserts the structural
 *  fingerprint of `events` against a live re-run of `inputFixtureName`. */
export function emitFixtureTest(
  events: TraceEvent[],
  opts: ToFixtureOptions,
): string {
  const fingerprint = fingerprintTrace(events);
  const fixturesImport =
    opts.fixturesImport ?? '../reducer/__fixtures__/trades-fixtures';
  const inputName = opts.inputFixtureName;
  const captured = JSON.stringify(fingerprint, null, 2);

  return [
    '/**',
    ' * Auto-generated by `intent-trace fixturize`. Asserts the structural',
    ` * fingerprint of \`${inputName}\` against a fresh reducer run.`,
    ' *',
    ' * Re-run the tool to refresh this fixture after intentional changes',
    ' * to pass behaviour. Do not edit by hand.',
    ' */',
    "import { describe, expect, test } from 'bun:test';",
    "import { reduceToIntent } from '../reducer/index';",
    "import { createInMemoryLogger } from '../logger';",
    `import { ${inputName} } from '${fixturesImport}';`,
    "import type { CorrelationId } from '../types';",
    '',
    'import { fingerprintTrace, type FixtureEvent } from \'../../../tools/intent-trace/src/to-fixture\';',
    "import { parseTrace } from '../../../tools/intent-trace/src/parse';",
    '',
    'const CAPTURED: FixtureEvent[] = ' + captured + ';',
    '',
    `describe('regression: ${inputName} structural fingerprint', () => {`,
    `  test('reducer run reproduces the captured event sequence', async () => {`,
    '    const logger = createInMemoryLogger();',
    `    await reduceToIntent(${inputName}.input, ${inputName}.grammar, {`,
    "      logger,",
    "      correlationId: 'corr-replay' as CorrelationId,",
    '    });',
    '    const jsonl = logger.events.map((e) => JSON.stringify(e)).join(\'\\n\');',
    '    const live = fingerprintTrace(parseTrace(jsonl));',
    '    expect(live).toEqual(CAPTURED);',
    '  });',
    '});',
    '',
  ].join('\n');
}

/** End-to-end: parse a JSONL string, optionally filter by correlationId,
 *  and produce a regression-test TS source string. */
export function jsonlToFixtureTest(jsonl: string, opts: ToFixtureOptions): string {
  const all = parseTrace(jsonl);
  const groups = groupByCorrelation(all);
  let events: TraceEvent[] | undefined;
  if (opts.correlationId) {
    events = groups.get(opts.correlationId);
    if (!events) {
      throw new Error(
        `jsonlToFixtureTest: no events found for correlationId '${opts.correlationId}'`,
      );
    }
  } else {
    events = groups.values().next().value;
    if (!events) {
      throw new Error('jsonlToFixtureTest: trace is empty');
    }
  }
  return emitFixtureTest(events, opts);
}
