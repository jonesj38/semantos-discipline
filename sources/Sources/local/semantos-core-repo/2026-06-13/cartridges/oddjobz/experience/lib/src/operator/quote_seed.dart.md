import 'conversation_turn.dart';
import 'quote_catalog.dart';
import 'quote_document.dart';

/// Build an editable Markdown quote draft from job conversation context.
///
/// The conversation is the source of truth: explicit prices mentioned by the
/// customer/operator become line items, and operator-configured catalog entries
/// whose words appear in the job conversation are inferred as additional draft
/// lines. Every autogenerated line carries provenance refs so regenerated
/// quotes can be traced back to exact conversation patches/turns and catalog
/// policy entries.
QuoteDocument quoteDraftSeededFromConversation({
  required String jobId,
  required Iterable<ConversationTurn> turns,
  Iterable<QuoteCatalogItem> catalogItems = const [],
  DateTime? now,
}) {
  final timestamp = now ?? DateTime.now().toUtc();
  final turnList = turns.toList(growable: false);
  final items = <QuoteLineItem>[];
  final contextNotes = <String>[];

  var moneyMentions = 0;
  var catalogMatches = 0;

  for (final turn in turnList) {
    final text = turn.bodyText.trim();
    if (text.isEmpty) continue;
    if (!turn.isInbound && turn.participantRole != 'external') continue;

    final money = _firstMoneyMentionCents(text);
    if (money == null) continue;

    final snippet = _snippet(text);
    final turnRef = _turnRef(turn);
    items.add(
      QuoteLineItem(
        description: snippet,
        quantity: 1,
        unitCents: money,
        unit: 'ea',
        category: 'conversation',
        sourceCatalogItemId: 'conversation:${turn.turnId}',
        provenanceRefs: [turnRef],
      ),
    );
    contextNotes.add('Conversation seed (${turn.surface}, $turnRef): $snippet');
    moneyMentions += 1;
  }

  for (final item in inferCatalogLineItemsFromConversation(
    turns: turnList,
    catalogItems: catalogItems,
    existing: items,
  )) {
    items.add(item);
    contextNotes.add(
      'Catalog match: ${item.description} (${item.provenanceRefs.join(', ')})',
    );
    catalogMatches += 1;
  }

  final summary = items.isEmpty
      ? ''
      : catalogMatches == 0
      ? 'Draft seeded from $moneyMentions conversation money mention${moneyMentions == 1 ? '' : 's'}.'
      : 'Draft autogenerated from $moneyMentions conversation price${moneyMentions == 1 ? '' : 's'} and $catalogMatches catalog match${catalogMatches == 1 ? '' : 'es'}.';

  return QuoteDocument(
    id: 'draft-${timestamp.microsecondsSinceEpoch}',
    jobId: jobId,
    status: 'draft',
    lineItems: items,
    paymentTerms: 'Due on completion.',
    notes: contextNotes.join('\n'),
    customerSummary: summary,
    markdown: quoteMarkdownFor(
      jobId: jobId,
      summary: summary,
      lineItems: items,
      paymentTerms: 'Due on completion.',
      notes: contextNotes,
    ),
    createdAt: timestamp,
    updatedAt: timestamp,
  );
}

List<QuoteLineItem> inferCatalogLineItemsFromConversation({
  required Iterable<ConversationTurn> turns,
  required Iterable<QuoteCatalogItem> catalogItems,
  Iterable<QuoteLineItem> existing = const [],
}) {
  final turnList = turns
      .where((turn) => turn.bodyText.trim().isNotEmpty)
      .toList(growable: false);
  if (turnList.isEmpty) return const [];
  final existingCatalogIds = {
    for (final item in existing)
      if (item.sourceCatalogItemId != null) item.sourceCatalogItemId,
  };
  final inferred = <QuoteLineItem>[];
  for (final item in catalogItems) {
    if (existingCatalogIds.contains(item.id)) continue;
    final tokens = _catalogTokens(item);
    if (tokens.isEmpty) continue;
    final matchingTurnRefs = <String>[];
    for (final turn in turnList) {
      final haystack = _normalise(turn.bodyText);
      if (tokens.any((token) => haystack.contains(token))) {
        matchingTurnRefs.add(_turnRef(turn));
      }
    }
    if (matchingTurnRefs.isEmpty) continue;
    inferred.add(
      item.toLineItem().copyWith(
        category: item.category,
        provenanceRefs: ['catalog:${item.id}', ...matchingTurnRefs],
      ),
    );
  }
  return inferred;
}

String quoteMarkdownFor({
  required String jobId,
  required String summary,
  required List<QuoteLineItem> lineItems,
  required String paymentTerms,
  Iterable<String> notes = const [],
}) {
  final b = StringBuffer()
    ..writeln('# Quote for $jobId')
    ..writeln();
  if (summary.trim().isNotEmpty) {
    b
      ..writeln(summary.trim())
      ..writeln();
  }
  b
    ..writeln('## Line items')
    ..writeln();
  if (lineItems.isEmpty) {
    b.writeln('- _No line items yet._');
  } else {
    for (final item in lineItems) {
      final desc = item.description.trim().isEmpty
          ? 'Untitled item'
          : item.description.trim();
      final provenance = item.provenanceRefs.isEmpty
          ? ''
          : ' _(source: ${item.provenanceRefs.join(', ')})_';
      b.writeln(
        '- $desc — ${_formatQty(item.quantity)} × ${formatCents(item.unitCents)} = ${formatCents(item.totalCents)}$provenance',
      );
    }
  }
  final total = lineItems.fold<int>(0, (sum, item) => sum + item.totalCents);
  b
    ..writeln()
    ..writeln('**Total: ${formatCents(total)}**')
    ..writeln();
  if (paymentTerms.trim().isNotEmpty) {
    b
      ..writeln('## Payment terms')
      ..writeln(paymentTerms.trim())
      ..writeln();
  }
  final cleanNotes = notes
      .map((n) => n.trim())
      .where((n) => n.isNotEmpty)
      .toList();
  if (cleanNotes.isNotEmpty) {
    b.writeln('## Source notes');
    for (final note in cleanNotes) {
      b.writeln('- $note');
    }
  }
  return b.toString().trimRight();
}

int? _firstMoneyMentionCents(String text) {
  final match = RegExp(
    r'(?:\$|AUD\s*)\s*([0-9][0-9,]*(?:\.[0-9]{1,2})?)',
    caseSensitive: false,
  ).firstMatch(text);
  if (match == null) return null;
  final raw = match.group(1)!.replaceAll(',', '');
  final dollars = double.tryParse(raw);
  if (dollars == null) return null;
  return (dollars * 100).round();
}

String _turnRef(ConversationTurn turn) => 'turn:${turn.turnId}';

String _snippet(String text) {
  final normalised = text.replaceAll(RegExp(r'\s+'), ' ').trim();
  if (normalised.length <= 96) return normalised;
  return '${normalised.substring(0, 93)}…';
}

String _normalise(String text) =>
    text.toLowerCase().replaceAll(RegExp(r'[^a-z0-9]+'), ' ').trim();

List<String> _catalogTokens(QuoteCatalogItem item) {
  final raw = [item.description, item.category].join(' ');
  return _normalise(
    raw,
  ).split(' ').where((token) => token.length >= 4).toSet().toList();
}

String _formatQty(double qty) =>
    qty == qty.roundToDouble() ? qty.toInt().toString() : qty.toString();
