// gen-edge-kat.ts — Cross-language KAT generator for the PWA contacts-PKI
// edge flow (invite → bilateral edge → BRC-69 backup recipe).
//
// Emits the canonical (TS/brain) invite-token + edge-creation vectors so
// the pure-Dart PWA implementation (apps/semantos/lib/src/wallet/
// edge_invite.dart) can assert BYTE-FOR-BYTE equality in
// test/wallet/edge_kat_test.dart. Equality proves a PWA-created edge
// interoperates with a brain-created one.
//
// Vectors are generated against the SAME functions the brain runs:
//   src/peer-invite.ts  — encodeInviteToken / buildInviteUrl
//   src/ecdh-edge.ts    — deriveEdgeSharedSecret / acceptInvite
//                         (edgeId + buildEdgeBackupRecipe via acceptInvite)
//
// acceptInvite() persists via local-edge-store.ts → localStorage, which
// doesn't exist under bun; we install a tiny in-memory shim so the REAL
// acceptInvite path (incl. its edgeId computation) is exercised, not a
// reimplementation.
//
// Run (writes the committed fixture):
//   cd cartridges/wallet-headers/brain
//   bun run scripts/gen-edge-kat.ts > \
//     ../../../apps/semantos/test/wallet/edge_kat.json
//
// Deterministic: all inputs are fixed; re-running reproduces the same JSON.

// ── localStorage shim (acceptInvite → saveEdgeEnvelope needs it) ──
const _ls = new Map<string, string>();
(globalThis as unknown as { localStorage: Storage }).localStorage = {
  getItem: (k: string) => (_ls.has(k) ? _ls.get(k)! : null),
  setItem: (k: string, v: string) => void _ls.set(k, v),
  removeItem: (k: string) => void _ls.delete(k),
  clear: () => _ls.clear(),
  key: (i: number) => Array.from(_ls.keys())[i] ?? null,
  get length() {
    return _ls.size;
  },
} as Storage;

import * as secp from '@noble/secp256k1';
import { encodeInviteToken, buildInviteUrl, type PeerInvite } from '../src/peer-invite';
import { deriveEdgeSharedSecret, acceptInvite } from '../src/ecdh-edge';

function hex(b: Uint8Array): string {
  let s = '';
  for (const x of b) s += x.toString(16).padStart(2, '0');
  return s;
}

function fromHex(h: string): Uint8Array {
  const out = new Uint8Array(h.length / 2);
  for (let i = 0; i < out.length; i++) out[i] = parseInt(h.slice(i * 2, i * 2 + 2), 16);
  return out;
}

// ── fixed inputs ──
// My identity (valid secp256k1 scalar 0x0102..0x20 < N) — the accepter.
const mySkHex = '0102030405060708090a0b0c0d0e0f101112131415161718191a1b1c1d1e1f20';
// The peer's (inviter's) identity key — a distinct valid scalar.
const theirSkHex = '00112233445566778899aabbccddeeff0102030405060708090a0b0c0d0e0f10';
const myCertId = 'a1b2c3d4e5f6a7b8c9d0e1f2a3b4c5d6';
const theirCertId = 'f0e1d2c3b4a5968778695a4b3c2d1e0f';
const nonce = '0f1e2d3c4b5a69788796a5b4c3d2e1f00112233445566778899aabbccddeeff00';
const timestamp = 1717000000000; // fixed so the invite token is deterministic
const INDICES = [0, 1, 7, 1234];

const mySk = fromHex(mySkHex);
const theirSk = fromHex(theirSkHex);
const myPk = secp.getPublicKey(mySk, true); // 33-byte compressed
const theirPk = secp.getPublicKey(theirSk, true);
const theirPkHex = hex(theirPk);

// The invite a peer hands me (inviter = "them").
const invite: PeerInvite = {
  certId: theirCertId,
  publicKey: theirPkHex,
  nonce,
  timestamp,
};

// Invite token + URL — pure (no time/randomness): exact byte target.
const inviteVector = {
  invite,
  token: encodeInviteToken(invite),
  url: buildInviteUrl(invite),
};

// Edge creation per index — drive the REAL acceptInvite so edgeId +
// backupRecipe come from the brain's own code path.
const myIdentity = { certId: myCertId, sk: mySk, pk: myPk };
const edgeVectors = INDICES.map((index) => {
  const env = acceptInvite(invite, myIdentity, index);
  if (!env) throw new Error(`acceptInvite returned null at index ${index}`);
  const sharedSecret = deriveEdgeSharedSecret(mySk, theirPk, index);
  if (!sharedSecret) throw new Error(`deriveEdgeSharedSecret null at index ${index}`);
  return {
    index,
    edgeId: env.edgeId,
    sharedSecretHex: hex(sharedSecret),
    backupRecipe: env.backupRecipe,
    theirCertId: env.theirCertId,
    theirPublicKey: env.theirPublicKey,
    edgeType: env.edgeType,
  };
});

const vectors = {
  _comment:
    'Cross-language KAT — brain (TS) contacts-PKI invite + bilateral edge. ' +
    'Generated by cartridges/wallet-headers/brain/scripts/gen-edge-kat.ts. ' +
    'Proves the PWA edge_invite.dart produces byte-identical invite tokens, ' +
    'edgeIds, ECDH shared secrets, and BRC-69 backup recipes.',
  mySkHex,
  myCertId,
  theirPkHex,
  theirCertId,
  nonce,
  timestamp,
  invite: inviteVector,
  edges: edgeVectors,
};

console.log(JSON.stringify(vectors, null, 2));
