// BRC-100 envelope parser/builder conformance tests.

import { describe, expect, test } from 'bun:test';
import * as secp from '@noble/secp256k1';
import { hmac } from '@noble/hashes/hmac';
import { sha256 as nobleSha256 } from '@noble/hashes/sha2';

import { buildEnvelope, parseEnvelope, hexToBytes } from '../src/brc100';

secp.etc.hmacSha256Sync = (key: Uint8Array, ...msgs: Uint8Array[]): Uint8Array =>
  hmac(nobleSha256, key, secp.etc.concatBytes(...msgs));

function freshKeypair(): { sk: Uint8Array; pk: Uint8Array } {
  const sk = secp.utils.randomPrivateKey();
  const pk = secp.getPublicKey(sk, true);
  return { sk, pk };
}

describe('BRC-100 envelope', () => {
  test('build → parse round-trip succeeds', () => {
    const { sk, pk } = freshKeypair();
    const body = new TextEncoder().encode(JSON.stringify({ method: 'getPublicKey' }));
    const env = buildEnvelope(sk, pk, body);
    expect(env.identityKey.length).toBe(33 * 2);
    expect(env.nonce.length).toBe(32 * 2);

    const parsed = parseEnvelope(env);
    expect(parsed.ok).toBe(true);
    if (parsed.ok) {
      expect(parsed.envelope.identityKey).toEqual(hexToBytes(env.identityKey));
      expect(parsed.envelope.timestamp).toBe(env.timestamp);
    }
  });

  test('rejects tampered body', () => {
    const { sk, pk } = freshKeypair();
    const body = new TextEncoder().encode('{"method":"getPublicKey"}');
    const env = buildEnvelope(sk, pk, body);
    // Tamper after signing.
    const bad = { ...env, body: new TextEncoder().encode('{"method":"createSignature"}') };
    const parsed = parseEnvelope(bad);
    expect(parsed.ok).toBe(false);
  });

  test('rejects timestamp outside replay window', () => {
    const { sk, pk } = freshKeypair();
    const body = new Uint8Array(0);
    const env = buildEnvelope(sk, pk, body, { now: () => 1000 });
    const parsed = parseEnvelope(env, { now: () => 100000, replayWindowSecs: 60 });
    expect(parsed.ok).toBe(false);
    if (!parsed.ok) expect(parsed.reason).toContain('timestamp outside');
  });

  test('rejects malformed envelopes structurally', () => {
    expect(parseEnvelope(null).ok).toBe(false);
    expect(parseEnvelope(42).ok).toBe(false);
    expect(parseEnvelope({}).ok).toBe(false);
    expect(
      parseEnvelope({
        identityKey: 'not-hex',
        nonce: 'not-hex',
        timestamp: 0,
        signature: 'not-hex',
        body: '',
      }).ok,
    ).toBe(false);
  });

  test('signature must come from identityKey', () => {
    const a = freshKeypair();
    const b = freshKeypair();
    const body = new Uint8Array([1, 2, 3]);
    const env = buildEnvelope(a.sk, a.pk, body);
    // Lie about the identity key — same signature, different pubkey.
    const tampered = { ...env };
    let bHex = '';
    for (const byte of b.pk) bHex += byte.toString(16).padStart(2, '0');
    tampered.identityKey = bHex;
    const parsed = parseEnvelope(tampered);
    expect(parsed.ok).toBe(false);
  });
});
