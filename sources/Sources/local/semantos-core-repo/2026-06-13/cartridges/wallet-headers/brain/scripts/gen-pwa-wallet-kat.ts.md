// gen-pwa-wallet-kat.ts — Cross-language KAT generator for the PWA ↔ brain
// wallet unification (L11 P6).
//
// Emits the canonical (TS/brain) change + cell-anchor key vectors for a fixed
// (identityKey, index) and (identityKey, typeHash, index) so the pure-Dart PWA
// wallet (apps/semantos) can assert byte-for-byte priv+pub equality in
// test/wallet/brain_wallet_kat_test.dart.
//
// The vectors are generated against the SAME functions the brain uses on
// mainnet: src/ecdh42.ts deriveChangeSk + src/cell-anchor.ts deriveCellAnchorSk.
// Equality of the PWA's `change`/`anchor` derivation with these proves the
// operator's PWA and brain derive the same on-chain keys — given the stated
// assumption that PWA cert_body == brain identitySk (same operator key).
//
// Run (writes the committed fixture):
//   cd cartridges/wallet-headers/brain
//   bun run scripts/gen-pwa-wallet-kat.ts > \
//     ../../../apps/semantos/test/wallet/brain_wallet_kat.json
//
// Deterministic: no randomness. Re-running reproduces the same JSON.

import * as secp from '@noble/secp256k1';
import { deriveChangeSk } from '../src/ecdh42';
import { deriveCellAnchorSk } from '../src/cell-anchor';

function hex(b: Uint8Array): string {
  let s = '';
  for (const x of b) s += x.toString(16).padStart(2, '0');
  return s;
}

function fromHex(h: string): Uint8Array {
  const out = new Uint8Array(h.length / 2);
  for (let i = 0; i < out.length; i++) out[i] = parseInt(h.slice(i * 2, i * 2 + 2), 16);
  return out;
}

// Fixed operator identity key (a valid secp256k1 scalar: 0x0102..0x20, < N).
const identityKeyHex = '0102030405060708090a0b0c0d0e0f101112131415161718191a1b1c1d1e1f20';
// Fixed 32-byte cell type_hash.
const typeHashHex = '00112233445566778899aabbccddeeff0102030405060708090a0b0c0d0e0f10';

const identitySk = fromHex(identityKeyHex);
const typeHash = fromHex(typeHashHex);

const CHANGE_INDICES = [0, 1, 47];
const ANCHOR_INDICES = [0, 1, 5];

function changeVector(index: number) {
  const sk = deriveChangeSk(identitySk, index);
  if (!sk) throw new Error(`deriveChangeSk returned null at index ${index}`);
  const pk = secp.getPublicKey(sk, true);
  return { index, privHex: hex(sk), pubHex: hex(pk) };
}

function anchorVector(index: number) {
  const sk = deriveCellAnchorSk(identitySk, typeHash, index);
  if (!sk) throw new Error(`deriveCellAnchorSk returned null at index ${index}`);
  const pk = secp.getPublicKey(sk, true);
  return { index, privHex: hex(sk), pubHex: hex(pk) };
}

const vectors = {
  _comment:
    'Cross-language KAT — brain (TS) wallet change + cell-anchor keys. ' +
    'Generated by cartridges/wallet-headers/brain/scripts/gen-pwa-wallet-kat.ts. ' +
    'Assumption: PWA cert_body == brain identitySk.',
  identityKeyHex,
  typeHashHex,
  // changeProtocolHash = SHA256("BRC-42-wallet-change")[0:16]
  // anchorProtocolHash = SHA256(hex(typeHash))[0:16]
  change: CHANGE_INDICES.map(changeVector),
  anchor: ANCHOR_INDICES.map(anchorVector),
};

console.log(JSON.stringify(vectors, null, 2));
