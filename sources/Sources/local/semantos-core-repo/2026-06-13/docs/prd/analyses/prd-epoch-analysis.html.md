// Helper: epoch bar chart in a card
function epochBars(epochs, maxM) {
  const mx = maxM || Math.max(...Object.values(epochs).map(e=>e.mentions), 1);

  }).join('');
}

// Panel: Flow — horizontal bar flow for all concepts
function buildFlow() {
  const panel = document.getElementById('panel-flow');
  let html = `
    <div class="section-title">Concept Flow Diagram</div>
    <div class="section-desc">
      Every concept shown as three bars: mention volume in each epoch.
      Bar width is proportional to mentions (log scale for readability).
      Sorted by total volume across all epochs.
    </div>
  `;

  // Get all concepts sorted by total mentions
  const allConcepts = Object.entries(D.migrations)
    .map(([concept, data]) => ({
      concept,
      ...data,
      total: Object.values(data.epochs).reduce((s,e) => s + e.mentions, 0)
    }))
    .filter(c => c.total > 0)
    .sort((a,b) => b.total - a.total);

  const maxTotal = Math.max(...allConcepts.map(c => Math.max(...Object.values(c.epochs).map(e=>e.mentions))), 1);
  const scale = m => Math.max(0, Math.log(m+1)/Math.log(maxTotal+1) * 180);

  allConcepts.forEach(c => {
    const s = c.epochs['shomee-alpha'];
    const e = c.epochs['semantic-seed'];
    const r = c.epochs['semantos-core'];
    html += `<div class="flow-row">
      <div class="flow-label">${c.concept}</div>
      <div class="flow-bar" style="width:${scale(s.mentions)}px;background:var(--shomee)">${s.mentions>10?s.mentions:''}</div>
      <div class="flow-arrow">→</div>
      <div class="flow-bar" style="width:${scale(e.mentions)}px;background:var(--seed)">${e.mentions>10?e.mentions:''}</div>
      <div class="flow-arrow">→</div>
      <div class="flow-bar" style="width:${scale(r.mentions)}px;background:var(--core)">${r.mentions>10?r.mentions:''}</div>
      <div class="flow-type"><span class="badge ${c.migration_type}">${c.migration_type}</span></div>
    </div>`;
  });

  panel.innerHTML = html;
}

function buildTerms() {
  const panel = document.getElementById('panel-terms');
  let html = `
    <div class="section-title">Epoch DNA — Distinguishing TF-IDF Terms</div>
    <div class="section-desc">
      The top terms that distinguish each epoch from the others (TF-IDF weighted).
      This is the linguistic fingerprint of each era.
    </div>
    <div class="terms-grid">
  `;

  ['shomee-alpha','semantic-seed','semantos-core'].forEach(epoch => {
    const cls = epoch === 'shomee-alpha' ? 'shomee' : epoch === 'semantic-seed' ? 'seed' : 'core';
    const terms = D.epoch_similarity.distinguishing_terms[epoch] || [];
    html += `<div class="terms-col">
      <div class="terms-header ${cls}">${epoch}</div>`;
    terms.forEach(t => {
      html += `<div class="term-row"><span>${t.term}</span><span class="w">${t.weight.toFixed(4)}</span></div>`;
    });
    html += '</div>';
  });

  html += `</div>

    <div class="section-title">Epoch Similarity Matrix</div>
    <div class="section-desc">Cosine similarity between epoch-level TF-IDF vectors. Lower similarity to semantos-core means more architectural divergence from the original codebase.</div>
    <div class="sim-matrix">
      <div class="sim-cell header"></div>
      <div class="sim-cell header" style="color:var(--shomee)">shomee-alpha</div>
      <div class="sim-cell header" style="color:var(--seed)">semantic-seed</div>
      <div class="sim-cell header" style="color:var(--core)">semantos-core</div>
  `;

  const sim = D.epoch_similarity.similarity_matrix;
  ['shomee-alpha','semantic-seed','semantos-core'].forEach(a => {
    const color = EPOCH_COLORS[a];
    html += `<div class="sim-cell label" style="color:${color}">${a}</div>`;
    ['shomee-alpha','semantic-seed','semantos-core'].forEach(b => {
      const v = sim[a][b];
      const bg = v > 0.7 ? 'rgba(52,211,153,0.15)' : v > 0.5 ? 'rgba(251,191,36,0.15)' : 'rgba(248,113,113,0.1)';
      html += `<div class="sim-cell" style="background:${bg};font-weight:600">${v.toFixed(3)}</div>`;
    });
  });

  html += '</div>';
  panel.innerHTML = html;
}
