#!/usr/bin/env bun
/**
 * glossary-to-md — render `docs/canon/glossary.yml` into the textbook
 * Appendix A glossary section.
 *
 * Usage:
 *   bun docs/canon/render/glossary-to-md.ts                 # stdout
 *   bun docs/canon/render/glossary-to-md.ts > out.md
 *   bun docs/canon/render/glossary-to-md.ts --format=table  # MD table form
 *
 * Stage: minimal-but-working stub. Reads the canon, sorts entries by id,
 * emits a markdown definition list. Empty canon yields a "no entries
 * yet" placeholder so the pipeline is testable from day one.
 */

import { readFileSync } from 'node:fs';
import { resolve } from 'node:path';
import { fileURLToPath } from 'node:url';
import { dirname } from 'node:path';
import { parse as parseYaml } from 'yaml';

interface GlossaryEntry {
  id: string;
  canonical: string | null;
  aliases: string[];
  definition: string | null;
  short?: string;
  sources?: string[];
  notes?: string;
  related?: string[];
}

interface GlossaryDoc {
  entries: GlossaryEntry[];
}

const HERE = dirname(fileURLToPath(import.meta.url));
const CANON_DIR = resolve(HERE, '..');
const GLOSSARY_FILE = resolve(CANON_DIR, 'glossary.yml');

const argv = process.argv.slice(2);
const format = argv.find((a) => a.startsWith('--format='))?.split('=')[1] ?? 'list';

function load(): GlossaryDoc {
  const raw = readFileSync(GLOSSARY_FILE, 'utf-8');
  const parsed = parseYaml(raw) as GlossaryDoc | null;
  if (!parsed || typeof parsed !== 'object') {
    return { entries: [] };
  }
  return { entries: parsed.entries ?? [] };
}

function renderEntry(e: GlossaryEntry): string {
  const term = e.canonical ?? e.id;
  const aliasLine =
    e.aliases.filter((a) => a !== term).length > 0
      ? `  *Also known as:* ${e.aliases.filter((a) => a !== term).map((a) => `\`${a}\``).join(', ')}.`
      : null;
  const def = e.definition ?? e.short ?? '_(definition pending — canonical not yet decided)_';
  const sources = e.sources?.length ? `\n  *Source:* ${e.sources.map((s) => `\`${s}\``).join(', ')}.` : '';
  return [
    `**${term}** (\`${e.id}\`)`,
    `  ${def}`,
    aliasLine,
    sources,
  ]
    .filter(Boolean)
    .join('\n');
}

function renderTable(entries: GlossaryEntry[]): string {
  const rows = entries.map((e) => {
    const term = e.canonical ?? e.id;
    const aliases = e.aliases.filter((a) => a !== term).join(', ') || '—';
    const def = (e.definition ?? e.short ?? '_pending_').replace(/\n/g, ' ').replace(/\|/g, '\\|');
    return `| \`${e.id}\` | ${term} | ${aliases} | ${def} |`;
  });
  return [
    '| ID | Canonical | Aliases | Short |',
    '|---|---|---|---|',
    ...rows,
  ].join('\n');
}

function emit(): string {
  const doc = load();
  const sorted = [...doc.entries].sort((a, b) => a.id.localeCompare(b.id));

  const header = [
    '# Appendix A — Glossary',
    '',
    '> Rendered from `docs/canon/glossary.yml`. Do not edit this file directly —',
    '> edit the YAML and re-run `bun docs/canon/render/glossary-to-md.ts`.',
    '',
  ].join('\n');

  if (sorted.length === 0) {
    return [
      header,
      '_No glossary entries yet — `docs/canon/glossary.yml` is empty._',
      '',
      'See `docs/canon/README.md` for the schema and the next-session fill plan.',
      '',
    ].join('\n');
  }

  const body = format === 'table' ? renderTable(sorted) : sorted.map(renderEntry).join('\n\n');

  const summary = [
    '',
    '---',
    '',
    `_${sorted.length} entries; ${sorted.filter((e) => e.canonical).length} with canonical decided._`,
    '',
  ].join('\n');

  return header + body + summary;
}

const output = emit();
process.stdout.write(output);
process.stdout.write('\n');
